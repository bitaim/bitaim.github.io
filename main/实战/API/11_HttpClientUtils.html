/**
 * http调用工具类
 */
public class HttpClientUtils {

    // 日志
    private static Logger logger = LoggerFactory.getLogger(HttpClientUtils.class);

    // 连接池管理
    private static final PoolingHttpClientConnectionManager connectionPoolManager = new PoolingHttpClientConnectionManager();

    static {
        // 设置最大连接数
        connectionPoolManager.setMaxTotal(10);
        // 设置主机地址的最大连接数
        connectionPoolManager.setDefaultMaxPerRoute(5);
    }

    private static RequestConfig requestConfig = null;

    static {
        // 设置建立连接超时时间 响应超时时间
        requestConfig = RequestConfig.custom().setConnectTimeout(2000).setSocketTimeout(2000).build();
    }

    private static ConnectionKeepAliveStrategy keepAliveStrategy = new ConnectionKeepAliveStrategy() {
        @Override
        public long getKeepAliveDuration(HttpResponse response, HttpContext context) {
            HeaderElementIterator it = new BasicHeaderElementIterator
                    (response.headerIterator(HTTP.CONN_KEEP_ALIVE));
            while (it.hasNext()) {
                HeaderElement he = it.nextElement();
                String param = he.getName();
                String value = he.getValue();
                if (value != null && param.equalsIgnoreCase
                        ("timeout")) {
                    return Long.parseLong(value) * 1000;
                }
            }
            return 60 * 1000;//如果没有约定，则默认定义时长为60s
        }
    };

    // httpclient是一个线程安全的类 全局保留一个即可
    private static CloseableHttpClient httpClient = HttpClients.custom()
            .setConnectionManager(connectionPoolManager)
            .setKeepAliveStrategy(keepAliveStrategy)
            .setDefaultRequestConfig(requestConfig)
            .build();


    /**
     * Application/json POST请求
     *
     * @param requestUrl  请求地址
     * @param paramObject 请求体
     * @param headersMap  请求头
     */
    public static JSONObject doPost(@NonNull String requestUrl, @NonNull JSONObject paramObject, @Nullable Map<String, String> headersMap) {
        if (StringUtils.isBlank(requestUrl) || paramObject == null) {
            throw new IllegalArgumentException(String.format("参数异常，requestUrl:{ %s },paramObject:{ %s }", requestUrl, paramObject));
        }
        HttpPost httpPost = new HttpPost(requestUrl);
        StringEntity entity = new StringEntity(paramObject.toString(), "UTF-8");
        entity.setContentEncoding("UTF-8");
        entity.setContentType("application/json");
        httpPost.setEntity(entity);
        if (headersMap != null && headersMap.size() > 0) {
            headersMap.forEach((key, value) -> {
                httpPost.setHeader(key, value);
            });
        }
        JsonObjectResponseHandler jsonObjectResponseHandler = new JsonObjectResponseHandler();
        JSONObject jsonObject = null;
        try {
            jsonObject = httpClient.execute(httpPost, jsonObjectResponseHandler);
        } catch (IOException e) {
            e.printStackTrace();
        }
        return jsonObject;
    }

    public static void doPost(@NonNull String requestUrl, @NonNull JSONObject paramObject) {
        doPost(requestUrl, paramObject, null);
    }

    public static void main(String[] args) throws InterruptedException {
        LoginRequestVO loginRequest = new LoginRequestVO();
        loginRequest.setAction("login");
        for (int i = 0; i < 10; i++) {
            new Thread(() -> {
                doPost("http://localhost:8080/pbl/User", JSONObject.parseObject(JSONObject.toJSONString(loginRequest)));
            }, "线程[" + i + "]").start();
        }
        Thread.sleep(50000);
    }
}

import com.alibaba.fastjson.JSONObject;
import org.apache.http.HttpEntity;
import org.apache.http.impl.client.AbstractResponseHandler;
import org.apache.http.util.EntityUtils;

import java.io.IOException;

public class JsonObjectResponseHandler extends AbstractResponseHandler<JSONObject> {

    @Override
    public JSONObject handleEntity(HttpEntity entity) throws IOException {
        String responseString = EntityUtils.toString(entity, "utf-8");
        EntityUtils.consume(entity);
        return JSONObject.parseObject(responseString);
    }

}
