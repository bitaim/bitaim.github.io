    # Curator配置

      Curator是一款apache开源的框架，对zookeeper进行了封装，简化了开发成本。

    1、引入相关依赖

    <dependency>
    <groupId>org.apache.curator</groupId>
    <artifactId>curator-recipes</artifactId>
    <version>4.3.0</version>
</dependency>

      curator-recipes依赖中已经包含zookeeper依赖和curator-framework，一些博客什么的都是胡乱粘贴，在maven的依赖中很明了的引入了。

      特别说明，目前curator-recipes最新版本已经是5.x，由于未找到相关文档，5.x版本引入了新的监听机制，将以前的标记为过期了。故使用4.x最新版本，也就是4.3.0版本。

    2、客户端相关指令

    public class CuratorClientTest {

    public static void main(String[] args) throws Exception {
        // 重试策略 必需
        RetryPolicy retryPolicy = new ExponentialBackoffRetry(1000, 2);
        CuratorFramework client = CuratorFrameworkFactory.builder()
                .connectString("127.0.0.1:2181") //如果是集群 用 “ , ” 隔开
                .retryPolicy(retryPolicy)
                .sessionTimeoutMs(30 * 1000) // 会话超时时间
                .build();
        // 开启连接
        client.start();

        // 创建(永久)节点
        client.create().forPath("/test","测试".getBytes(CharsetUtil.UTF_8));
        // 获取值
        byte[] bytes = client.getData().forPath("/test");
        String value = new String(bytes, CharsetUtil.UTF_8);
        System.out.println(value);
        // 创建临时节点 creatingParentsIfNeeded 创建多级 
        // /server是永久节点 imServer_1是临时节点 客户端断开连接即消失
        client.create().creatingParentsIfNeeded().withMode(CreateMode.EPHEMERAL).forPath("/server/imServer_1", "127.0.0.1:8080".getBytes());
        // 获取子节点
        List<String> childPathList = client.getChildren().forPath("/");
        // 赋值（修改）
        Stat stat = client.setData().forPath("/server", "127.0.0.1:8080".getBytes());
        // 删除节点 如果/test节点下还有节点 则删除报错
        client.delete().forPath("/test");
        // 删除多级节点
        client.delete().deletingChildrenIfNeeded().forPath("/test");

        // 获取节点数据
        Stat stat = new Stat();
        // 同时存储节点的状态
        client.getData().storingStatIn(stat).forPath("/server");
        // 如果在客户端连接期间 其他客户端已经修改了该节点数据 则更新失败
        // 与CAS机制相同
        // 如果version为-1，或者不带withVersion()方法则直接更新 
        client.setData().withVersion(stat.getVersion()).forPath("/server", "TEST".getBytes());
    }
}

    3、监听节点变化

    public class CuratorWatchTest {
    public static void main(String[] args) throws Exception {
        RetryPolicy retryPolicy = new ExponentialBackoffRetry(1000, 2);
        CuratorFramework client = CuratorFrameworkFactory.builder()
                .connectString("127.0.0.1:2181")
                .retryPolicy(retryPolicy)
                .sessionTimeoutMs(5 * 1000)
                .build();

        client.start();

        // 客户端 节点路径 不压缩
        NodeCache nodeCache = new NodeCache(client, "/server", false);
        nodeCache.start();
        // 监听节点的变化
        nodeCache.getListenable().addListener(new NodeCacheListener() {
            @Override
            public void nodeChanged() throws Exception {
                ChildData currentData = nodeCache.getCurrentData();
                if (currentData == null) {
                    System.out.println("节点不存在或者已被删除");
                } else {
                    System.out.println("路径：" + currentData.getPath());
                    if (currentData.getData() == null) {
                        System.out.println("数据：{}");
                    } else {
                        System.out.println("数据：{" + new String(currentData.getData()) + "}");
                    }
                }
            }
        });
        client.close();
    }
}

    # 二次封装分布式锁

      每一个项目相当于一个zookeeper客户端，所以只需要在SpringBoot配置一个CuratorFramework即可。配置类如下。

    @Configuration
public class CuratorFrameworkConfig{

    private CuratorFramework client;

    @Bean(value = "curatorClient")
    public CuratorFramework createCuratorFramework() {
        // 重试之间的间隔时间 和 重试最大次数
        RetryPolicy retryPolicy = new ExponentialBackoffRetry(1 * 1000, 2);
        client = CuratorFrameworkFactory.builder()
                // 可配置多个地址，通过,分隔
                .connectString("127.0.0.1:2181")
                .retryPolicy(retryPolicy)
                // 当client连接上server 就会一直向server发送心跳包 来维护会话的有效
                // 当超过1秒 没有发送心跳 则可能会网络问题掉线或者是client宕机
                // 触发重试策略
                .sessionTimeoutMs(2 * 1000)
                .build();
        client.start();
        return client;
    }

     /**
     * 项目停止时销毁Bean
     */
    @PreDestroy
    public void destroy() {
        client.close();
    }

}

      对Curaor框架提供的分布式锁进行了二次封装，使其对某一个资源进行加锁，而不是针对于接口级别，使加锁更细致。

    @Service
public class InterProcessMutexPlus {

    @Autowired
    private CuratorFramework curatorClient;

    private static final String GANG = "/";

    public InterProcessMutexPlus() {}

    private Map<String, InterProcessMutex> map = new HashMap<>(16);

    private ReentrantLock lock = new ReentrantLock();

    public InterProcessMutex getInterProcessMutex(String path) {
        checkPath(path);
        // 采用double check检测 防止创建多个实例
        InterProcessMutex interProcessMutex = map.get(path);
        if (interProcessMutex == null) {
            try {
                lock.lock();
                if (interProcessMutex == null) {
                    interProcessMutex = new InterProcessMutex(curatorClient, path);
                    map.put(path, interProcessMutex);
                }
            } finally {
                lock.unlock();
            }
        }
        return interProcessMutex;
    }

    public void lock(String source) throws Exception {
        InterProcessMutex interProcessMutex = getInterProcessMutex(GANG + source);
        interProcessMutex.acquire();
    }

    public boolean lock(String source, long timeout) throws Exception {
        InterProcessMutex interProcessMutex = getInterProcessMutex(GANG + source);
        return interProcessMutex.acquire(timeout, TimeUnit.SECONDS);
    }

    public void unLock(String source) throws Exception {
        InterProcessMutex interProcessMutex = getInterProcessMutex(GANG + source);
        interProcessMutex.release();
    }

    /**
     * 校验参数合理性，拓展
     */
    private void checkPath(String source) {
        if (source == null || source.length() == 0) {
            throw new IllegalArgumentException("不合法资源");
        }
    }

}

      简单的使用，代码如下。

    @RestController
@RequestMapping(value = "dispatcher")
public class DispatcherController {

    @Autowired
    private InterProcessMutexPlus interProcessMutexPlus;

    @GetMapping(value = "lock")
    public String lock(String source) throws Exception {
        System.out.println(Thread.currentThread().getName());
        boolean lockSuccess = false;
        try {
            // 在5秒内获取某个资源的锁，获取不到则返回false
            lockSuccess = interProcessMutexPlus.lock(source, 5);
            System.out.println(lockSuccess);
            if (lockSuccess) {
                System.out.println("执行业务逻辑");
                Thread.sleep(10000);
                return "success";
            }
            System.out.println("获取锁超时失败");
            return "false";
        } finally {
            if (lockSuccess) {
                interProcessMutexPlus.unLock(source);
            }
        }
    }
}
