# 字符串

字符串使用 "+" 拼接时，只要存在一个引用，那么生成的对象就在堆中 。 字符串使用 concat 拼接时，生成的对象都在堆中。

public class StringTest {

public static void main(String[] args) {

// // 只会在堆中创建String对象 // String a = new String("123"); // // 字符串常量池如果有该字符串，则直接返回字符串常量池相应的地址 // 字符串常量池如果没有该字符串，先将堆中字符串放入字符串常量池，然后返回字符串常量池相应的地址 // String b = a.intern(); // // a 是堆中的引用 b是常量池中的引用 因此为false // System.out.println(a == b);

// String s = new String("1"); // String t = new String("1"); // String x = new String("1") + new String("1"); // 生成在堆中，而不是字符串常量池 // String s1 = s.intern(); // String s2 = "1"; // String s3 = "11"; // System.out.println(s == s1); // false // System.out.println(s1 == s2); // true // System.out.println(s == t); // false // System.out.println(s.intern() == t.intern()); // true // System.out.println(x == s3); // false // System.out.println(x.intern() == s3.intern()); // true

    // x 指向堆
    String x = new String("1") + new String("1");
    // x1 指向堆
    String x1 = new String("1") + "1";
    String x2 = "1" + "1";
    String s3 = "11";

    System.out.println(x == s3);  // false
    System.out.println(x1 == s3);  // false
    System.out.println(x2 == s3); // true




}

}
