    package com.bitaim.study.code.generator;

import com.baomidou.mybatisplus.annotation.DbType;
import com.baomidou.mybatisplus.generator.AutoGenerator;
import com.baomidou.mybatisplus.generator.config.DataSourceConfig;
import com.baomidou.mybatisplus.generator.config.GlobalConfig;
import com.baomidou.mybatisplus.generator.config.PackageConfig;
import com.baomidou.mybatisplus.generator.config.StrategyConfig;
import com.baomidou.mybatisplus.generator.config.rules.DateType;
import com.baomidou.mybatisplus.generator.config.rules.NamingStrategy;
import com.bitaim.study.code.util.FileUtil;


public class CodeGenerator {

    /**
     * 表名列表
     */
    private static String[] tableNames = new String[]{"user", "menu"};

    /**
     * 代码生成目录
     */
    private static String codeGeneratorCatalogue = "E:\\踏踏实实学技术\\code-generator\\src\\main\\java";

    private static String parentFolder = "com.bitaim.study";

    public static void main(String[] args) {

        AutoGenerator autoGenerator = new AutoGenerator();
        autoGenerator.setGlobalConfig(getGlobalConfig());
        autoGenerator.setDataSource(getDataSourceConfig());
        autoGenerator.setStrategy(getStrategyConfig());
        autoGenerator.setPackageInfo(getPackageConfig());
        // 生成代码
        autoGenerator.execute();

        deleteFolder("service");
        deleteFolder("controller");

    }

    private static GlobalConfig getGlobalConfig() {
        GlobalConfig globalConfig = new GlobalConfig();
        globalConfig
                .setAuthor("云聪")
                .setOutputDir(codeGeneratorCatalogue)           // 代码生成目录
                .setFileOverride(true)                          // 允许代码覆盖
                .setDateType(DateType.ONLY_DATE)
                .setEntityName("%sDO")
                .setBaseResultMap(true)
                .setBaseColumnList(true);
        return globalConfig;
    }

    private static DataSourceConfig getDataSourceConfig() {
        DataSourceConfig dataSourceConfig = new DataSourceConfig();
        dataSourceConfig
                .setDbType(DbType.MYSQL)
                .setDriverName("com.mysql.jdbc.Driver")
                .setUrl("jdbc:mysql://127.0.0.1:3306/test?useUnicode=true&characterEncoding=utf-8&useSSL=false&serverTimeZone=Asia/Shanghai")
                .setUsername("root")
                .setPassword("mysql");
        return dataSourceConfig;
    }

    private static StrategyConfig getStrategyConfig() {
        StrategyConfig strategyConfig = new StrategyConfig();
        strategyConfig
                .setNaming(NamingStrategy.underline_to_camel)
                .setEntityLombokModel(true)
                .setRestControllerStyle(true)
                .setInclude(tableNames);     // 表名列表
        return strategyConfig;
    }

    private static PackageConfig getPackageConfig() {
        PackageConfig packageConfig = new PackageConfig();
        packageConfig
                .setParent(parentFolder)
                .setEntity("model")
                .setMapper("mapper")
                .setXml("mapper");
        return packageConfig;
    }

    private static void deleteFolder(String folderName) {
        String filePath = codeGeneratorCatalogue + "\\" + parentFolder.replace(".", "\\") + "\\" + folderName;
        FileUtil.deleteFile(filePath);
    }

}
