# CompletableFuture

import lombok.extern.slf4j.Slf4j;
import org.junit.Test;

import java.util.concurrent.CompletableFuture;

@Slf4j
public class CompletableFutureTest {

    /**
     * 异步任务
     */
    @Test
    public void aysncTask() {

        CompletableFuture<Integer> cf = CompletableFuture.supplyAsync(() -> {
            return 10;
        });

        System.out.println(cf.join());

        cf = CompletableFuture.supplyAsync(() -> {
            int number = 10 / 0;
            return 10;
        }).exceptionally(throwable -> {
            log.error("异步任务执行失败，原因：{}", throwable.getMessage(), throwable);
            return null;
        });

        System.out.println(cf.join());

    }

    /**
     * 任务回调
     */
    @Test
    public void taskCallback() {

        // thenApply 接收上一个任务的结果，有返回值
        CompletableFuture<Integer> cf = CompletableFuture.supplyAsync(() -> {
            return 10;
        }).thenApply(result -> {
            return result * 3;
        });
        System.out.println(cf.join());

        // thenAccept 接收上一个任务的结果，没有返回值
        CompletableFuture.supplyAsync(() -> {
            return 10;
        }).thenAccept(result -> {
            System.out.println(result * 3);
        });

    }

    /**
     * 合并两个异步任务的结果
     */
    @Test
    public void taskCombile() {

        CompletableFuture<Integer> cf1 = CompletableFuture.supplyAsync(() -> {
            return 10;
        });

        CompletableFuture<String> cf2 = CompletableFuture.supplyAsync(() -> {
            return "20";
        });

        // 将两个异步任务结果归并，并形成新的结果集
        CompletableFuture<Integer> cf3 = cf1.thenCombine(cf2, (result1, result2) -> {
            // result1:10, result2:20
            log.info("result1:{}, result2:{}", result1, result2);
            return result1 + Integer.parseInt(result2);
        });

        System.out.println(cf3.join());

    }

    @Test
    public void lotsTaskCombine() {

        CompletableFuture<Integer> cf1 = CompletableFuture.supplyAsync(() -> {
            ThreadUtil.sleep(50);
            return 10;
        });

        CompletableFuture<String> cf2 = CompletableFuture.supplyAsync(() -> {
            ThreadUtil.sleep(500);
            return "20";
        });

        CompletableFuture<Integer> cf3 = CompletableFuture.supplyAsync(() -> {
            ThreadUtil.sleep(200);
            return 10;
        });

        CompletableFuture<String> cf4 = CompletableFuture.supplyAsync(() -> {
            ThreadUtil.sleep(300);
            return "20";
        });

        // 阻塞主线程，等待4个任务全部完成
        CompletableFuture.allOf(cf1, cf2, cf3, cf4);

        Integer result = cf1.join() + Integer.parseInt(cf2.join()) + cf3.join() + Integer.parseInt(cf4.join());
        System.out.println(result);
        
    }

     /**
     * 选择最快的方案
     */
    @Test
    public void getBestMethod() {

        CompletableFuture<Integer> cf1 = CompletableFuture.supplyAsync(() -> {
            ThreadUtil.sleep(150);
            return 150;
        });

        CompletableFuture<Integer> cf2 = CompletableFuture.supplyAsync(() -> {
            ThreadUtil.sleep(5000);
            return 5000;
        });
        CompletableFuture<Integer> cf3 = CompletableFuture.supplyAsync(() -> {
            ThreadUtil.sleep(50);
            return 50;
        });

        CompletableFuture<Integer> cf4 = CompletableFuture.supplyAsync(() -> {
            ThreadUtil.sleep(500);
            return 500;
        });

        CompletableFuture<Object> finalCf = CompletableFuture.anyOf(cf1, cf2, cf3, cf4).thenApply(result -> {
            return result;
        });

        System.out.println(finalCf.join());

    }

}

# 小案例

  多线程调用下游服务。

public class CfTest {

    private static ThreadPoolExecutor threadPoolExecutor = new ThreadPoolExecutor(4, 8, 30, TimeUnit.SECONDS, new ArrayBlockingQueue<>(1000));

    @Test
    public void test01() {
        List<String> userIds = new ArrayList<>();
        userIds.add("1111");
        userIds.add("2222");
        CompletableFuture[] completableFutures = userIds.stream().map(userId -> CompletableFuture.supplyAsync(() -> getUserById(userId), threadPoolExecutor)
                .whenComplete((result, exception) -> {

                })).toArray(CompletableFuture[]::new);
        CompletableFuture.allOf(completableFutures);
        
        for (CompletableFuture completableFuture : completableFutures) {
            User user = (User)completableFuture.join() ;
            System.out.println(user);
        }
    }

    private static User getUserById(String userId) {
        User user = new User();
        user.setUsername(userId);
        return user;
    }

}
