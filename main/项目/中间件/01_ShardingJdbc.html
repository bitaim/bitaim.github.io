# 标准分片策略

  标准分片策略之精准分片算法。

import org.apache.shardingsphere.api.sharding.standard.PreciseShardingAlgorithm;
import org.apache.shardingsphere.api.sharding.standard.PreciseShardingValue;

import java.util.Collection;

/**
 * 标准分片策略 精准分片算法
 */
public class DatabasePreciseShardingAlgorithm implements PreciseShardingAlgorithm<String> {

    /**
     * @param dataSources   数据源集合。
     *                      分库时，值为所有分片库的集合。例如：["ds0","ds1","ds2"]
     *                      分表时，值为对应分片库中所有分片表的集合。例如：["user_0","user_1","user_2"]
     * @param shardingValue 分片属性。包括 logicTableName逻辑表，columnName分片键，value为从SQL中解析出的分片键的值
     * @return
     */
    @Override
    public String doSharding(Collection<String> dataSources, PreciseShardingValue<String> shardingValue) {

        // 获取短链码首位（首位是库位）
        String codePrefix = shardingValue.getValue().substring(0, 1);

        for (String dataSource : dataSources) {
            // 获取具体的库位后缀
            String targetNameSuffix = dataSource.substring(dataSource.length() - 1);
            if (codePrefix.equals(targetNameSuffix)) {
                return dataSource;
            }

        }

        return null;

    }

}
