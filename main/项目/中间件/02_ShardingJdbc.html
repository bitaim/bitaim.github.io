# ShardingJdbc

  ShardingJdbc作为最常用的分库分表中间件，实现了SQL拦截、SQL改写、SQL路由、结果归并等功能。

# 分片策略

# Inline Strategy

  根据单一分片键进行精确分片，无需实现分片算法。比如 以id作为分片键，适用的SQL如下：

insert into user values(?,?,?,?);

select * from user where id = ?;
-- Inline Strategy 对 in查询 的解析方式是不太精确的，可能会命中多个分片。
select * from user where id in (?, ?);

  无法对分片键进行范围查询，会直接抛出异常。比如 between...and...、大于、小于等。

# Standard Strategy

  根据单一分片键进行精确或者范围分片，需要实现分片算法。

# Complex Strategy

  根据多个分片键进行精确或者范围分片，需要实现分片算法。实际使用场景很少，几乎不用。

# Hint Strategy

  使用与SQL无关的方式进行分片，不需要根据分片键进行路由，通过自定义算法强制路由。

# 分片算法

  实现分片算法，需要将分片键类型作为泛型传入接口。

# 精准分片算法

  根据自身业务场景，实现 PreciseShardingAlgorithm 接口即可实现精准分片算法。

/**
 * Precise sharding algorithm.
 * 
 * @param <T> class type of sharding value
 */
public interface PreciseShardingAlgorithm<T extends Comparable<?>> extends ShardingAlgorithm {
    
    /**
     * Sharding.
     * 
     * @param availableTargetNames available data sources or tables's names
     * @param shardingValue sharding value
     * @return sharding result for data source or table's name
     */
    String doSharding(Collection<String> availableTargetNames, PreciseShardingValue<T> shardingValue);
}

# 范围分片算法

  根据自身业务场景，实现 RangeShardingAlgorithm 接口即可实现范围分片算法。

  可以看到方法返回值可以是多个分片，RangeShardingValue 可以获取到上下限的具体值。

/**
 * Range sharding algorithm.
 * 
 * @param <T> class type of sharding value
 */
public interface RangeShardingAlgorithm<T extends Comparable<?>> extends ShardingAlgorithm {
    
    /**
     * Sharding.
     * 
     * @param availableTargetNames available data sources or tables's names
     * @param shardingValue sharding value
     * @return sharding results for data sources or tables's names
     */
    Collection<String> doSharding(Collection<String> availableTargetNames, RangeShardingValue<T> shardingValue);
}

# Hint分片算法

  根据自身业务场景，实现 HintShardingAlgorithm 接口即可实现Hint分片算法。

/**
 * Sharding algorithm for hint without sharding keys.
 * 
 * @param <T> class type of sharding value
 */
public interface HintShardingAlgorithm<T extends Comparable<?>> extends ShardingAlgorithm {
    
    /**
     * Sharding.
     * 
     * <p>sharding value injected by hint, not in SQL.</p>
     *
     * @param availableTargetNames available data sources or tables's names
     * @param shardingValue sharding value
     * @return sharding result for data sources or tables's names
     */
    Collection<String> doSharding(Collection<String> availableTargetNames, HintShardingValue<T> shardingValue);
}

# 标准分片策略之精准分片算法

  在分片键中埋入库位和表位，进而实现分库分表。淘宝的订单号就按一定规则加入了库表位信息。

import org.apache.shardingsphere.api.sharding.standard.PreciseShardingAlgorithm;
import org.apache.shardingsphere.api.sharding.standard.PreciseShardingValue;

import java.util.Collection;

/**
 * 标准分片策略 精准分片算法
 */
public class DatabasePreciseShardingAlgorithm implements PreciseShardingAlgorithm<String> {

    /**
     * @param dataSources   数据源集合。
     *                      分库时，值为所有分片库的集合。例如：["ds0","ds1","ds2"]
     *                      分表时，值为对应分片库中所有分片表的集合。例如：["user_0","user_1","user_2"]
     * @param shardingValue 分片属性。包括 logicTableName逻辑表，columnName分片键，value为从SQL中解析出的分片键的值
     * @return
     */
    @Override
    public String doSharding(Collection<String> dataSources, PreciseShardingValue<String> shardingValue) {

        // 获取短链码首位（首位是库位）
        String codePrefix = shardingValue.getValue().substring(0, 1);

        for (String dataSource : dataSources) {
            // 获取具体的库位后缀
            String targetNameSuffix = dataSource.substring(dataSource.length() - 1);
            if (codePrefix.equals(targetNameSuffix)) {
                return dataSource;
            }

        }

        return null;

    }

}

# 拓展接口

  ShardingSphere提供了大量基于SPI的扩展接口。比如：ShardingKeyGenerator主键生成接口，其内置实现有SnowflakeShardingKeyGenerator和UUIDShardingKeyGenerator。

  通过SPI机制就可以读取到该接口的多个实现类，我们也可以基于这种方式拓展自己的主键生成策略。
