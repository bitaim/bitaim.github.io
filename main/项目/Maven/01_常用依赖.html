public class JSONObjectTest {

    public static void main(String[] args) {

        User user = new User();
        user.setUserId(111);
        user.setRealName("云聪");

        System.out.println(JSONObject.toJSONString(user)); // json中忽略值为null字段
        System.out.println(JSONObject.toJSONString(user, SerializerFeature.WriteMapNullValue)); // json中保留值为null字段
        
    }

}
