# SQL命令

  SQL尽量与业务分开，不要出现一个查询业务就写一个SQL，要尽力保证SQL的复用性。

  SQL写的越简单越好，不要试图用一个SQL实现特别复杂的功能，不利于维护和修改。

# 增

public interface CouponUseRecordMapper {

    void insert(CouponUseRecordDO record);

    void insertBatch(@Param(value = "records") List<CouponUseRecordDO> records);

}

  下面两个SQL主键策略使用自增主键，如果有实现的主键生成策略，去掉keyProperty="id"和useGeneratedKeys="true"。

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.bitaim.dream.coupon.mapper.CouponUseRecordMapper">

    <resultMap id="baseResultMap" type="com.bitaim.dream.coupon.domain.CouponUseRecordDO">
        <id column="id" property="id"/>
        <result column="out_trade_no" property="outTradeNo"/>
        <result column="coupon_record_id" property="couponCollectRecordId"/>
        <result column="lock_status" property="lockStatus"/>
        <result column="created_time" property="createdTime"/>
        <result column="updated_time" property="updatedTime"/>
    </resultMap>
    
    <insert id="insert" keyProperty="id" useGeneratedKeys="true" parameterType="com.bitaim.dream.coupon.domain.CouponUseRecordDO">
        INSERT INTO coupon_use_record(`out_trade_no`, `coupon_collect_record_id`, `lock_status`, `created_time`, `updated_time`)
        VALUES( #{outTradeNo}, #{couponCollectRecordId}, #{lockStatus}, #{createdTime}, #{updatedTime})
    </insert>

    <insert id="insertBatch" keyProperty="id" useGeneratedKeys="true" parameterType="com.bitaim.dream.coupon.domain.CouponUseRecordDO">
        INSERT INTO
        coupon_use_record(`out_trade_no`, `coupon_collect_record_id`, `lock_status`, `created_time`, `updated_time`)
        VALUES
        <foreach collection="records" item="record" separator=",">
            (
              #{record.outTradeNo}, 
              #{record.couponCollectRecordId}, 
              #{record.lockStatus},
              #{record.createdTime}, 
              #{record.updatedTime}
            )
        </foreach>
    </insert>

</mapper>

public interface MenuMapper {

    void deleteByIds(@Param(value = "ids") List<Integer> ids);

}

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.bitaim.dream.user.mapper.MenuMapper">

    <resultMap id="baseResultMap" type="com.bitaim.dream.user.domain.MenuDO">
        <id column="id" property="id"/>
        <result column="parent_id" property="parentId"/>
        <result column="id_link_chain" property="idLinkChain"/>
        <result column="menu_name" property="menuName"/>
        <result column="menu_remarks" property="menuRemarks"/>
        <result column="menu_access_path" property="menuAccessPath"/>
        <result column="menu_level" property="menuLevel"/>
        <result column="menu_type" property="menuType"/>
        <result column="hidden_status" property="hiddenStatus"/>
        <result column="created_time" property="createdTime"/>
        <result column="updated_time" property="updatedTime"/>
    </resultMap>

    <delete id="deleteById">
        DELETE FROM `menu` WHERE id = #{id}
    </delete>
    
    <delete id="deleteByIds">
        DELETE FROM `menu` WHERE id in 
        <foreach collection="ids" item="id" open="(" separator="," close=")">
            #{id}
        </foreach>
    </delete>

</mapper>

public interface MenuMapper {

    void updateById(MenuDO menu);

}

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.bitaim.dream.user.mapper.MenuMapper">

    <resultMap id="baseResultMap" type="com.bitaim.dream.user.domain.MenuDO">
        <id column="id" property="id"/>
        <result column="parent_id" property="parentId"/>
        <result column="id_link_chain" property="idLinkChain"/>
        <result column="menu_name" property="menuName"/>
        <result column="menu_remarks" property="menuRemarks"/>
        <result column="menu_access_path" property="menuAccessPath"/>
        <result column="menu_level" property="menuLevel"/>
        <result column="menu_type" property="menuType"/>
        <result column="hidden_status" property="hiddenStatus"/>
        <result column="created_time" property="createdTime"/>
        <result column="updated_time" property="updatedTime"/>
    </resultMap>

    <update id="updateById" parameterType="com.bitaim.dream.user.domain.MenuDO">
        UPDATE
        `user`
        <set>
            <if test="menuName != null and menuName.length > 0">
                menu_name = #{menuName},
            </if>
            <if test="menuRemarks != null and menuRemarks.length > 0">
                menu_remarks = #{menuRemarks},
            </if>
            <if test="menuAccessPath != null">
                menu_access_path = #{menuAccessPath},
            </if>
            <if test="hiddenStatus != null">
                hidden_status = #{hiddenStatus},
            </if>
            updated_time = #{updatedTime}
        </set>
        WHERE id = #{id}
    </update>

</mapper>

# 查

public interface MenuMapper {

    MenuDO getById(Integer id);

    List<MenuDO> listMenus(MenuDO menu);

}

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.bitaim.dream.user.mapper.MenuMapper">

    <resultMap id="baseResultMap" type="com.bitaim.dream.user.domain.MenuDO">
        <id column="id" property="id"/>
        <result column="parent_id" property="parentId"/>
        <result column="id_link_chain" property="idLinkChain"/>
        <result column="menu_name" property="menuName"/>
        <result column="menu_remarks" property="menuRemarks"/>
        <result column="menu_access_path" property="menuAccessPath"/>
        <result column="menu_level" property="menuLevel"/>
        <result column="menu_type" property="menuType"/>
        <result column="hidden_status" property="hiddenStatus"/>
        <result column="created_time" property="createdTime"/>
        <result column="updated_time" property="updatedTime"/>
    </resultMap>
    
    <sql id="baseColumns">
        id, parent_id, id_link_chain, menu_name, menu_remarks, menu_access_path, menu_level, menu_type, hidden_status
    </sql>

    <select id="getById" resultMap="baseResultMap">
        SELECT
            <include refid="baseColumns"/>
        FROM
            `menu`
        WHERE
            id = #{id}
    </select>

    <select id="listMenus" resultMap="baseResultMap" parameterType="com.bitaim.dream.user.domain.MenuDO">
        SELECT
            <include refid="baseColumns"/>
        FROM
        `menu`
        WHERE
        menu_type = #{menuType}
        <if test="parentId != null">
            and parent_id = #{parentId}
        </if>
        <if test="idLinkChain != null and idLinkChain.length > 0">
            and id_link_chain LIKE CONCAT(#{idLinkChain}, '%')
        </if>
        and hidden_status = 0
    </select>
</mapper>

    <collection property="sourceList" ofType="com.walk.distribution.core.model.SourceDO">
        <id property="sourceId" column="source_id"/>
        <result property="source" column="source"/>
        <collection property="skuList" ofType="com.walk.distribution.core.model.SkuItemDO">
            <result property="sourceId" column="source_id"/>
            <result property="size" column="size"/>
        </collection>
    </collection>
</resultMap>
