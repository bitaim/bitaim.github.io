    # 注解激活式

      使用 @Import注解 注入 AliMessageServerConfiguration 配置类。

    import com.bitaim.spring.boot.starter.config.AliMessageServerConfiguration;
import org.springframework.context.annotation.Import;

import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

@Target({ElementType.TYPE})
@Retention(RetentionPolicy.RUNTIME)
@Import({AliMessageServerConfiguration.class})
public @interface EnableAliMessage {
}

    # 配置类

      使用 @EnableConfigurationProperties注解 激活参数类。

    import com.bitaim.spring.boot.starter.properties.AliMessageServerProperties;
import com.bitaim.spring.boot.starter.server.AliMessageServer;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
import org.springframework.boot.context.properties.EnableConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
@EnableConfigurationProperties(AliMessageServerProperties.class)
@ConditionalOnProperty(
        prefix = "ali.message",
        name = {"access-key-id","access-secret","sign-name"}
)
public class AliMessageServerConfiguration {

    @Autowired
    private AliMessageServerProperties aliMessageServerProperties;

    @Bean(value = "aliMessageServer")
    public AliMessageServer aliMessageServer() {
        AliMessageServer aliMessageServer = new AliMessageServer();
        aliMessageServer.wrapper(aliMessageServerProperties);
        return aliMessageServer;
    }

}

    # 业务类

      定义Starter的最终目的，封装组件到业务类。

    import com.alibaba.fastjson.JSONObject;
import com.aliyuncs.CommonRequest;
import com.aliyuncs.CommonResponse;
import com.aliyuncs.DefaultAcsClient;
import com.aliyuncs.IAcsClient;
import com.aliyuncs.exceptions.ClientException;
import com.aliyuncs.http.MethodType;
import com.aliyuncs.profile.DefaultProfile;
import com.bitaim.spring.boot.starter.bean.SendRequest;
import com.bitaim.spring.boot.starter.properties.AliMessageServerProperties;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class AliMessageServer {

    private static Logger logger = LoggerFactory.getLogger(AliMessageServer.class);

    private String region;

    private String accessKeyId;

    private String accessSecret;

    private String signName;

    public void wrapper(AliMessageServerProperties properties) {
        this.region = properties.getRegion();
        this.accessKeyId = properties.getAccessKeyId();
        this.accessSecret = properties.getAccessSecret();
        this.signName = properties.getSignName();
    }

    public boolean send(SendRequest request) {
        DefaultProfile profile = DefaultProfile.getProfile(region, accessKeyId, accessSecret);
        IAcsClient client = new DefaultAcsClient(profile);
        CommonRequest commonRequest = new CommonRequest();
        commonRequest.setSysMethod(MethodType.POST);
        commonRequest.setSysDomain("dysmsapi.aliyuncs.com");
        commonRequest.setSysVersion("2017-05-25");
        commonRequest.setSysAction("SendSms");
        commonRequest.putQueryParameter("RegionId", region);
        commonRequest.putQueryParameter("PhoneNumbers", request.getPhone());
        commonRequest.putQueryParameter("SignName", signName);
        commonRequest.putQueryParameter("TemplateCode", request.getTemplateCode());
        commonRequest.putQueryParameter("TemplateParam", String.format("{\"code\":\"%s\"}", request.getCode()));
        // 发送请求
        try {
            CommonResponse response = client.getCommonResponse(commonRequest);
            JSONObject jsonObject = JSONObject.parseObject(response.getData());
            String responseCode = jsonObject.getString("Code");
            if ("OK".equals(responseCode)) {
                return true;
            }
        } catch (ClientException e) {
            logger.error("AliMessageServiceImpl.sendCodeMessage()发送验证码异常，参数:{}，异常信息:{}", request, e.getMessage());
        }
        return false;
    }
}
