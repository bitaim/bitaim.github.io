    # 代码

      注解 @NotBlank，用于String类型的字段上，不允许值为null，不允许值为空串。

      注解 @NotEmpty，可用于String、Collection、数组，不允许值为null，不允许值的长度为0。

      注解 @NotNull，不允许值为null。

      注解 @Max和@Min，用于String、数字类型，指定值的数字范围。

      注解 @Length(min=1, max=10)，指定值的字符串长度范围。

      注解 @Size(min=1, max=10)和@NotNull，用于Collection、数组，指定值的长度范围。

    import lombok.Data;

import javax.validation.constraints.NotBlank;

@Data
public class User {

    @NotBlank(message = "用户名不能为空")
    private String username;

}

    # 使用方式一

    import com.bitaim.study.spring.enums.BizCode;
import com.bitaim.study.spring.exception.BizException;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.stereotype.Component;

import javax.validation.ConstraintViolation;
import javax.validation.Validation;
import javax.validation.Validator;
import javax.validation.ValidatorFactory;
import java.util.Set;

/**
 * 参数校验器
 */
@Component
public class ParameterValidator implements InitializingBean {

    private Validator validator;

    @Override
    public void afterPropertiesSet() throws Exception {
        ValidatorFactory validatorFactory = Validation.buildDefaultValidatorFactory();
        this.validator = validatorFactory.getValidator();
    }

    public void validate(Object obj) {
        if (obj == null) {
            return;
        }

        // 校验不通过，直接抛出异常
        Set<ConstraintViolation<Object>> validateResults = validator.validate(obj);
        if (validateResults.size() > 0) {
            throw new RuntimeException("参数不合法！");
        }

    }

}

    # 使用方式二

      在类上添加 @Validated 注解，在方法的入参上添加 @Valid注解。

    import com.bitaim.study.spring.vo.User;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.validation.annotation.Validated;
import javax.validation.Valid;

@Slf4j
@Service
@Validated
public class DemoService {

    @Override
    public String checkParameter(@Valid User user) {
        return null;
    }

}
