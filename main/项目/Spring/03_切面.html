    # 注解

    import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

/**
 * 接口耗时器注解
 */
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
public @interface InterfaceTimer {

    /**
     * 描述信息
     *
     * @return
     */
    String description() default "";

}

    # 切面类

    import com.bitaim.study.spring.constants.SignConstants;
import lombok.extern.slf4j.Slf4j;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.annotation.Pointcut;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;
import org.springframework.util.StringUtils;

import java.lang.reflect.Method;


/**
 * 接口耗时切面类
 */
@Slf4j
@Aspect
@Order(20)
@Component
public class InterfaceTimerAspect {

    private static String logTemplate = "方法 [%s] 执行耗时 [%s ms]";

    @Pointcut("@annotation(interfaceTimer)")
    public void pointCut(InterfaceTimer interfaceTimer) {
    }

    @Around(value = "pointCut(interfaceTimer)")
    public Object around(ProceedingJoinPoint proceedingJoinPoint, InterfaceTimer interfaceTimer) throws Throwable {
        Method method = this.getExecutedMethod(proceedingJoinPoint);
        Object executeResult;
        if (interfaceTimer != null) {
            long startTimeMillis = System.currentTimeMillis();
            executeResult = proceedingJoinPoint.proceed();
            long endTimeMillis = System.currentTimeMillis();
            String description = interfaceTimer.description();
            if (StringUtils.isEmpty(description)) {
                // 全限类名#方法名
                description = proceedingJoinPoint.getTarget().getClass().getName() + SignConstants.JING + method.getName();
            }
            log.info(String.format(logTemplate, description, (endTimeMillis - startTimeMillis)));
        } else {
            executeResult = proceedingJoinPoint.proceed();
        }
        return executeResult;
    }

    /**
     * 获取执行的方法
     *
     * @param joinPoint
     * @return
     */
    public Method getExecutedMethod(JoinPoint joinPoint) {
        Signature signature = joinPoint.getSignature();
        MethodSignature methodSignature = (MethodSignature) signature;
        return methodSignature.getMethod();
    }
    
}
