# AtomicStampedReference

  CAS执行操作中，可能会出现ABA问题。为了解决该问题，JDK中引入了AtomicStampedReference，引入了游标字段，也就是版本号，来解决ABA问题。

public class AtomicStampedReference<V> {

    /**
     * 内部静态类 Pair，负责将引用和游标（引用的版本号）包装起来
     */
    private static class Pair<T> {
        final T reference;
        final int stamp;
        private Pair(T reference, int stamp) {
            this.reference = reference;
            this.stamp = stamp;
        }
        static <T> Pair<T> of(T reference, int stamp) {
            return new Pair<T>(reference, stamp);
        }
    }

    // 存储数据
    private volatile Pair<V> pair;

    
    // AtomicStampedReference构造方法，并初始化 引用和游标
    public AtomicStampedReference(V initialRef, int initialStamp) {
        pair = Pair.of(initialRef, initialStamp);
    }

    // 返回当前引用
    public V getReference() {
        return pair.reference;
    }

    // 返回当前游标
    public int getStamp() {
        return pair.stamp;
    }

    // 传递一个int数组，是为了存储当前游标
    // 并返回当前引用
    public V get(int[] stampHolder) {
        Pair<V> pair = this.pair;
        //  将当前游标 赋值 给数组
        stampHolder[0] = pair.stamp;
        return pair.reference;
    }

    /**
     * 与 compareAndSet 方法调用相同，不清楚到底有何区别
     * 既然写了，肯定是有意义的，有人说JVM会对该方法进行特殊处理
     */
    public boolean weakCompareAndSet(V   expectedReference,
                                     V   newReference,
                                     int expectedStamp,
                                     int newStamp) {
        return compareAndSet(expectedReference, newReference,
                             expectedStamp, newStamp);
    }


    /**
     * 如果预期的引用和当前的引用相同，并且预期的游标与当前的游标相同。
     * 则可以执行 引用和游标的原子性的赋值操作
     */
    public boolean compareAndSet(V   expectedReference,
                                 V   newReference,
                                 int expectedStamp,
                                 int newStamp) {
        Pair<V> current = pair;
        return
            expectedReference == current.reference &&
            expectedStamp == current.stamp &&
            ((newReference == current.reference &&
              newStamp == current.stamp) ||
             casPair(current, Pair.of(newReference, newStamp)));
    }

    /**
     * 单纯的赋值操作
     */
    public void set(V newReference, int newStamp) {
        Pair<V> current = pair;
        if (newReference != current.reference || newStamp != current.stamp)
            this.pair = Pair.of(newReference, newStamp);
    }

    /**
     * 更新预期引用的游标
     */
    public boolean attemptStamp(V expectedReference, int newStamp) {
        Pair<V> current = pair;
        return
            expectedReference == current.reference &&
            (newStamp == current.stamp ||
             casPair(current, Pair.of(expectedReference, newStamp)));
    }

    // Unsafe mechanics

    private static final sun.misc.Unsafe UNSAFE = sun.misc.Unsafe.getUnsafe();
    // 获取 pair 变量 相对于 对象 在 内存中的偏移量
    private static final long pairOffset =
        objectFieldOffset(UNSAFE, "pair", AtomicStampedReference.class);

    // this 可以看作当前对象的内存地址
    // pairOffset 定位到内存地址中的偏移量
    // 将cmp 更新为 val
    private boolean casPair(Pair<V> cmp, Pair<V> val) {
        return UNSAFE.compareAndSwapObject(this, pairOffset, cmp, val);
    }

    static long objectFieldOffset(sun.misc.Unsafe UNSAFE,
                                  String field, Class<?> klazz) {
        try {
            return UNSAFE.objectFieldOffset(klazz.getDeclaredField(field));
        } catch (NoSuchFieldException e) {
            // Convert Exception to corresponding Error
            NoSuchFieldError error = new NoSuchFieldError(field);
            error.initCause(e);
            throw error;
        }
    }
}

import java.util.concurrent.atomic.AtomicStampedReference;

public class AtomicStampedReferenceTest {

    public static void main(String[] args) {

        String initString = "1";
        String update = "2";

        AtomicStampedReference<String> atomicStampedReference = new AtomicStampedReference<>(initString, 1);
        System.out.println(atomicStampedReference.getReference());
        System.out.println(atomicStampedReference.getStamp());

        boolean result = atomicStampedReference.compareAndSet(initString, update, 1, 2);

        int[] number = new int[1];
        String s = atomicStampedReference.get(number);
        System.out.println("引用的值为：" + s);
        System.out.println("当前的游标为：" + number[0]);
        System.out.println(result);
    }

}
