# MappedByteBuffer

  mmap是内存映射文件的方法，可以将一个文件映射到进程的内存地址空间，实现文件磁盘地址和进程的一段虚拟内存地址的映射。 实现这样的映射关系后，进程就可以采用指针的方式读写操作这一段内存。

  mmap操作的这一块内存可以理解为page cache，这一段内存是由操作系统管理的，操作系统会自动回写脏页到对应的文件磁盘上，完成了对文件的读写操作而不必再调用read，write等系统调用函数。 相反，内核空间对这段区域的修改也直接反应用户空间，从而实现不同进程间的文件共享。

  mmap和FileChannel都以缺页中断的方式，进行文件读写。以mmap读取一个1G的文件时首先会进行映射，映射是一个消耗极少的操作，此时并不意味着1G的文件被读入paga cache。

  只有在真正读写时，才会因为缺页中断从而在磁盘上加载文件到page cache中。

  mmap映射的过程可以理解为一个懒加载，只有get()时才会触发缺页中断。预读大小是由操作系统的算法决定的，可以默认为4kb。如果希望懒加载变成实时加载，可以按照4kb这个步长遍历该文件。

  MappedByteBuffer在用户空间的堆外内存，完成了虚拟内存地址与文件的磁盘地址的一一映射。

  当映射完文件后，并不会立即消耗真正的物理内存。只有在真正读写这些数据时，才会因为缺页异常而从磁盘上加载这些数据到内核空间的os cache中， 同时更新之前的映射表，将磁盘地址修改为真正的物理内存地址。 这么做的目的是，用户空间的一段虚拟内存地址与内核空间os cache的一段虚拟地址内存映射到相同的物理内存地址。

  基于这种原理的读写操作，省去了数据的拷贝，当读写时可以像操作内存一样去操作这个文件。

  当创建一个MappedByteBuffer后，磁盘上会立即生成一个同等大小的文件。

  MMAP这么强，那么FileChannel还有存在的意义吗？网上许多文章都说，MMAP操作大文件的性能比FileChannel高出一个数量级。但真正的情况呢？MMAP并非是文件IO的银弹，它只有在一次写入很小数据量的场景并且内存充足的情况下才能表现出比FileChannel稍微优异的性能。在Java中使用MappedByteBuffer还存在一些问题，如下：

  （1）MMAP使用时必须指定好内存映射的大小，并且一次性map的大小不能超过1.5G，重复map又会带来虚拟内存的回收、重新分配的问题，对于文件不确定大小的情形实在是太不友好了。

  （2）MMAP使用的是虚拟内存，和page cache一样是由操作系统来控制刷盘的，虽然可能通过force()来手动控制，但这个时间不好把握，在小内存场景下会很令人头疼。

  （3）MMAP的回收问题，当MappedByteBuffer不再需要时，可以手动释放占用的虚拟内存，但方法非常诡异。

  建议是，优先使用FileChannel完成初始代码的提交，在必须使用小数据量（比如几个字节）刷盘的场景下，再换成MMAP的实现。其他场景下，FileChannel完成可以cover，前提是要理解怎么合理使用FileChannel。

import java.io.File;
import java.io.IOException;
import java.io.RandomAccessFile;
import java.nio.MappedByteBuffer;
import java.nio.channels.FileChannel;

public class MappedByteBufferTest {

    public static void main(String[] args) throws IOException {

        String filePath = "D:/files";
        String fileName = "record.txt";

        FileChannel fileChannel = new RandomAccessFile(new File(filePath, fileName), "rw").getChannel();
        MappedByteBuffer mappedByteBuffer = fileChannel.map(FileChannel.MapMode.READ_WRITE, 0, fileChannel.size());

        // 步长4kb
        int step = 4 * 1024;

        // 预热文件
        for (long i = 0; i < fileChannel.size(); i += step) {
            mappedByteBuffer.get((int) i);
        }

    }

}

public class DirectByteBuffer {
    
    public ByteBuffer put(ByteBuffer src) {

        // 堆外内存
        if (src instanceof DirectByteBuffer) {
            if (src == this)
                throw new IllegalArgumentException();
            DirectByteBuffer sb = (DirectByteBuffer)src;

            int spos = sb.position();
            int slim = sb.limit();
            assert (spos <= slim);
            int srem = (spos <= slim ? slim - spos : 0);

            int pos = position();
            int lim = limit();
            assert (pos <= lim);
            int rem = (pos <= lim ? lim - pos : 0);

            if (srem > rem)
                throw new BufferOverflowException();
            
            // 调用 unsafe 的内存复制
            // sb.ix(spos)：堆外内存的地址
            // ix(pos)：映射文件（要写入的文件）的内存地址
            // (long)srem << 0：堆外内存数据的可读长度
            unsafe.copyMemory(sb.ix(spos), ix(pos), (long)srem << 0);
            sb.position(spos + srem);
            position(pos + srem);
        } else if (src.hb != null) {

            int spos = src.position();
            int slim = src.limit();
            assert (spos <= slim);
            int srem = (spos <= slim ? slim - spos : 0);

            put(src.hb, src.offset + spos, srem);
            src.position(spos + srem);

        } else {
            super.put(src);
        }
        return this;
    }
    
}


public class Unsafe {


    /**
    * 将指定的 内存地址块 复制到 另外一块内存地址块
    * 
    * srcBase：要拷贝数据的内存基地址（源地址）
    * srcOffset：源地址偏移量
    * destBase：要拷贝到的数据的内存基地址（目的地址）
    * destOffset：目的地址偏移量
    * bytes：要拷贝的数据长度
    */ 
    public native void copyMemory(Object srcBase, long srcOffset,
                                  Object destBase, long destOffset,
                                  long bytes);
    
    // 当传入的基地址为空时，两个偏移量地址（srcAddress、destAddress）要提供一个绝对地址
    public void copyMemory(long srcAddress, long destAddress, long bytes) {
        copyMemory(null, srcAddress, null, destAddress, bytes);
    }
    
}
