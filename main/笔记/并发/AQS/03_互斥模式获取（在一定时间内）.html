# 互斥模式获取（在一定时间内）

# 获取逻辑

public final boolean tryAcquireNanos(int arg,long nanosTimeout)
        throws InterruptedException {
    // 首先检查中断状态
    if (Thread.interrupted())
        throw new InterruptedException();
    // 获取失败，进入队列中等待
    return tryAcquire(arg) ||
        doAcquireNanos(arg, nanosTimeout);
}

  当前线程希望在自己阻塞的这段时间里，有人唤醒它，它去尝试获取锁。如果获取失败，判断剩余等于时间够不够，够的话继续阻塞，等待下一次被唤醒。还是获取不到，那就获取失败了/(ㄒoㄒ)/~~。

/**
 * 以独占模式在一定时间内获取
 * nanosTimeout：最大等待时间
 */
private boolean doAcquireNanos(int arg, long nanosTimeout)
        throws InterruptedException {
    if (nanosTimeout <= 0L)
        return false;
    // 计算获取的截至日志（单位：纳秒）
    // 1秒=1000毫秒 1毫秒=1000000纳秒
    final long deadline = System.nanoTime() + nanosTimeout;
    // 加入等待队列
    final Node node = addWaiter(Node.EXCLUSIVE);
    boolean failed = true;
    try {
        for (;;) {
            final Node p = node.predecessor();
            // 前置节点是头节点，则尝试获取
            if (p == head && tryAcquire(arg)) {
                setHead(node);
                p.next = null; // help GC
                failed = false;
                return true;
            }
            // 每进入一次循环，计算等待时间
            nanosTimeout = deadline - System.nanoTime();
            // 已经超过最大时间了，直接返回false
            if (nanosTimeout <= 0L)
                return false;
            // shouldParkAfterFailedAcquire() 返回true，
            // 代表前置节点的 waitStatus已经被设置为 -1 
            if (shouldParkAfterFailedAcquire(p, node) &&
                // 等待剩余时间 大于 1000纳秒（判断有没有必要在阻塞）
                nanosTimeout > spinForTimeoutThreshold) {

                // 以等待剩余时间，阻塞当前线程
                LockSupport.parkNanos(this, nanosTimeout);

            }
            // 线程被中断
            if (Thread.interrupted())
                // 抛出中断异常
                throw new InterruptedException();
        }
    } finally {
        // 获取锁超时，或者进入中断状态，取消当前节点
        if (failed)
            cancelAcquire(node);
    }
}

# 取消逻辑

# https://blog.csdn.net/qq_32099833/article/details/109786284

// 取消当前节点
private void cancelAcquire(Node node) {
    // Ignore if node doesn't exist
    if (node == null)
        return;
    // 清空线程
    node.thread = null;
    
    // 获取前置节点
    Node pred = node.prev;
    // 如果前置节点是取消状态
    while (pred.waitStatus > 0)
        // pred引用 一直向上移动（向上遍历）
        // 将当前节点的前置节点指向（👆） pred引用
        node.prev = pred = pred.prev;
    
    // 上述逻辑为了找到最近的非取消状态的节点，pred节点指向该节点
    
    Node predNext = pred.next;
    
    // 将当前节点更新为取消状态
    node.waitStatus = Node.CANCELLED;
    
    // 当前节点是尾节点，则更新尾节点到 pred节点
    if (node == tail && compareAndSetTail(node, pred)) {
        // 更新尾节点成功，清除后继节点
        compareAndSetNext(pred, predNext, null);
    } else {
        int ws;
        
        // 修改前置节点（pred节点）的状态置为-1，用来唤醒后继节点
        if (pred != head &&
            ((ws = pred.waitStatus) == Node.SIGNAL ||
             (ws <= 0 && compareAndSetWaitStatus(pred, ws, Node.SIGNAL))) &&
            pred.thread != null) {
            // 获取当前节点（要取消的节点）的后继节点
            Node next = node.next;
            
            if (next != null && next.waitStatus <= 0)
                // 将pred节点指向当前节点的后继节点，让当前节点出列
                compareAndSetNext(pred, predNext, next);
        } else {
            // 进入该逻辑的两种情况
            // 1、pred节点是头节点
            // 2、CAS修改pred节点状态为-1失败，导致节点出队失败，唤醒当前节点的后继节点（来完成出队操作）
            unparkSuccessor(node);
        }
        node.next = node; // help GC
    }
}

# 唤醒后逻辑

private boolean doAcquireNanos(int arg, long nanosTimeout)
        throws InterruptedException {
    if (nanosTimeout <= 0L)
        return false;
    final long deadline = System.nanoTime() + nanosTimeout;
    final Node node = addWaiter(Node.EXCLUSIVE);
    boolean failed = true;
    try {
        for (;;) {
            // 2、重新进入for循环
            final Node p = node.predecessor();
            // 3、不是头节点，或者获取锁失败
            if (p == head && tryAcquire(arg)) {
                setHead(node);
                p.next = null; // help GC
                failed = false;
                return true;
            }
            nanosTimeout = deadline - System.nanoTime();
            if (nanosTimeout <= 0L)
                return false;
            // 4、shouldParkAfterFailedAcquire 进入该方法
            // 跳过出队失败的前置节点（1），找到非取消的节点置为-1
            // 等待下一次唤醒我
            if (shouldParkAfterFailedAcquire(p, node) &&
                nanosTimeout > spinForTimeoutThreshold)
                // 1、突然被唤醒了
                LockSupport.parkNanos(this, nanosTimeout);
            if (Thread.interrupted())
                throw new InterruptedException();
        }
    } finally {
        if (failed)
            cancelAcquire(node);
    }
}
