    # 阻塞逻辑

    public final void await() throws InterruptedException {
    if (Thread.interrupted())
        throw new InterruptedException();
    // 将当前线程添加到等待队列
    Node node = addConditionWaiter();
    // 唤醒后继节点
    int savedState = fullyRelease(node);
    int interruptMode = 0;

    // （1）使用while，避免伪唤醒
    // （2）在同步队列中，代表已经被成功唤醒，跳过死循环
    while (!isOnSyncQueue(node)) {
        // 阻塞当前线程
        LockSupport.park(this);

        // 被唤醒后检测中断状态
        // interruptMode == 0 ，通过 signal 正常唤醒
        // interruptMode == -1 或者 1，由于中断而被唤醒，直接跳出死循环（已经加入到同步队列了无须while判断）
        if ((interruptMode = checkInterruptWhileWaiting(node)) != 0)
            break;
    }
    // 重新竞争锁
    // 在同步队列等待期间被中断，interruptMode == 0 或者 1
    if (acquireQueued(node, savedState) && interruptMode != THROW_IE) {
        // 中断模式置为1
        interruptMode = REINTERRUPT;
    }
        
    // 节点是由于中断而被唤醒，如果通过signal唤醒时会将当前节点出等待队列（nextWaiter为null）
    if (node.nextWaiter != null) {
        // 节点同时存在于等待队列和同步队列，清除等待队列中的中断节点
        unlinkCancelledWaiters();
    }
        
    if (interruptMode != 0) {
        // 将中断状态传播出去
        reportInterruptAfterWait(interruptMode);
    }
        
}

private void reportInterruptAfterWait(int interruptMode)
    throws InterruptedException {
    // 如果线程被中断，直接抛出异常
    if (interruptMode == THROW_IE)
        throw new InterruptedException();
    // 如果 == 1
    else if (interruptMode == REINTERRUPT)
        // 赋予线程中断标志位
        selfInterrupt();
}

    1、加入到等待队列

      加当前线程封装成Node节点，加入到Condition等待队列中。

    private Node addConditionWaiter() {
    // 获取等待队列的尾节点
    Node t = lastWaiter;
    // If lastWaiter is cancelled, clean out.
    // 如果尾节点不为空 并且 状态不为 CONDITION
    if (t != null && t.waitStatus != Node.CONDITION) {
        // 清理取消状态的等节点
        unlinkCancelledWaiters();
        t = lastWaiter;
    }
    // 封装成 Node新节点
    Node node = new Node(Thread.currentThread(), Node.CONDITION);
    // 尾节点为空
    if (t == null)
        // 头节点 指向 新节点
        firstWaiter = node;
    else
        // 添加到 等待队列尾部
        t.nextWaiter = node;
    // 尾节点 指向 新节点
    lastWaiter = node;
    return node;
}

      清理取消状态的节点。

    private void unlinkCancelledWaiters() {
    Node t = firstWaiter;
    // 负责存储 CONDITION状态的节点，在遍历过程中遇到取消的节点，直接指向下一个节点
    // 比如 A->B->C->D ，当前 trail->B，在遍历C的过程中发现C是取消状态，通过 trail.nextWaiter -> D，跳过取消状态的节点
    Node trail = null;
    while (t != null) {
        // 拿到下一次遍历的节点
        Node next = t.nextWaiter;
        // 当前节点的状态不为 CONDITION 状态
        if (t.waitStatus != Node.CONDITION) {
            // 断开该节点的后续连接
            t.nextWaiter = null;
            if (trail == null) {
                firstWaiter = next;
            } else {
                trail.nextWaiter = next;
            }
            // 遍历到最后一个节点时
            if (next == null) {
                // 重置尾节点
                lastWaiter = trail;
            }  
        } else {
            trail = t;
        }           
        t = next;
    }
}

    2、唤醒后继节点

    final int fullyRelease(Node node) {
    boolean failed = true;
    try {
        // 获取state值
        int savedState = getState();
        // Condition在使用过程中，都是需要拿到独占锁才可以操作的
        // 正常过程中都会释放成功，并且唤醒同步队列中的下一个节点
        if (release(savedState)) {
            failed = false;
            return savedState;
        } else {
            throw new IllegalMonitorStateException();
        }
    } finally {
        // 释放失败，将当前节点标记为取消状态
        // 重入锁导致释放失败???待考证
        if (failed)
            node.waitStatus = Node.CANCELLED;
    }
}

    3、检测中断

    // 当断当前节点是否在同步队列中
// 通过 await() 第一次进入时，肯定返回false
final boolean isOnSyncQueue(Node node) {
    // 节点的状态为 CONDITION，肯定还在等待队列中
    // 上一节点为null，肯定还在等待队列中，在同步队列的时候起码还有一个头节点
    if (node.waitStatus == Node.CONDITION || node.prev == null)
        return false;
    // 如果有后置节点 肯定在同步队列了（等待队列通过nextWaiter连接）
    if (node.next != null) {
        return true;
    }
    
    // （1）还没加入到等待队列，更别提加入到同步队列
    // （2）正在加入同步队列的过程中（竞争太大或者还没赋值next字段）
    // 从队尾向上寻找该节点
    return findNodeFromTail(node);
}


private int checkInterruptWhileWaiting(Node node) {
    // （1）线程未被中断，返回0
    // （2）线程被中断（清除中断状态），进入transferAfterCancelledWait(node逻辑)
    return Thread.interrupted() ? (transferAfterCancelledWait(node) ? THROW_IE : REINTERRUPT) : 0;
}

final boolean transferAfterCancelledWait(Node node) {
    // （1）interrupt 发生在 signal之前，返回true，外层返回 -1
    if (compareAndSetWaitStatus(node, Node.CONDITION, 0)) {
        // 将当前节点加入同步队列
        enq(node);
        return true;
    }

    // 上面if修改失败，很可能另外一个线程调用了 signal方法，已经将状态修改为 0
    // （2）中断发生在signal之后，返回false，外层返回 1
    // 此时node节点正在加入同步队列，通过自旋等待入同步队列完成即可
    while (!isOnSyncQueue(node))
        Thread.yield();
    return false;
}

    # 唤醒逻辑

    public final void signal() {
    // 当前线程非持有锁，直接抛出异常
    if (!isHeldExclusively())
        throw new IllegalMonitorStateException();
    // 拿到Condition条件队列的头节点
    Node first = firstWaiter;
    if (first != null)
        doSignal(first);
}

private void doSignal(Node first) {
    do {
        // 头节点指向下一个节点
        // 判断如果头节点为null
        if ( (firstWaiter = first.nextWaiter) == null) {
            // 清空尾节点
            lastWaiter = null;
        }
        // 断开引用（出等待队列成功）
        first.nextWaiter = null;
    // 唤醒失败，再次尝试直到不满足条件为止
    } while (!transferForSignal(first) &&
                (first = firstWaiter) != null);
}


final boolean transferForSignal(Node node) {
    /*
    * If cannot change waitStatus, the node has been cancelled.
    */
    // 如果更新 waitStatus值失败，代表节点已经被取消
    if (!compareAndSetWaitStatus(node, Node.CONDITION, 0))
        return false;

    /*
    * Splice onto queue and try to set waitStatus of predecessor to
    * indicate that thread is (probably) waiting. If cancelled or
    * attempt to set waitStatus fails, wake up to resync (in which
    * case the waitStatus can be transiently and harmlessly wrong).
    */
    // 将当前节点加入同步队列，返回上一个节点
    Node p = enq(node);
    int ws = p.waitStatus;
    // 直接唤醒线程的两个条件
    // （1）上一节点状态为取消
    // （2）将上一节点状态修改为-1失败
    if (ws > 0 || !compareAndSetWaitStatus(p, ws, Node.SIGNAL))
        // 唤醒当前节点的线程（被唤醒的线程还需要重新拿锁，拿不到又会在同步队列中阻塞，还可以跳过取消节点）
        LockSupport.unpark(node.thread);
    return true;
}
