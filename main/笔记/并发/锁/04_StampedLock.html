 * 返回的邮戳，可用于解锁或者转换模式
 */
public long writeLock() {
    long s, next;  
    // 获取state值与ABITS进行与运算，如果等于0。通过CAS更新state值（加128），如果更新成功，则返回。否则进入 acquireWrite方法，阻塞获取。
    return ((((s = state) & ABITS) == 0L &&

            next : acquireWrite(false, 0L));

    private volatile String token = "!23";

    private StampedLock stampedLock = new StampedLock();

    /**
     * 更新数据
     *
     * @param token
     */
    public void write(String token) {
        // 获取写锁
        long stamp = stampedLock.writeLock();
        try {
            this.token = token;
        } finally {
            // 解锁写锁
            stampedLock.unlockWrite(stamp);
        }
    }

    /**
     * 读取数据
     *
     * @return
     */
    public String readOp() {
        // 先获得乐观锁
        long optimisticRead = stampedLock.tryOptimisticRead();
        // 读取数据
        String current = this.token;
        // 校验邮戳失败（代表读取数据期间被其他线程修改，当前数据是不安全的）
        if (!stampedLock.validate(optimisticRead)) {
            // 获取读锁（也可以尝试再次获取乐观读）
            long stamp = stampedLock.readLock();
            try {
                // 重新读取数据
                current = this.token;
            } finally {
                // 解锁读锁
                stampedLock.unlockRead(stamp);
            }
        }
        return current;
    }


    /**
     * 读取数据
     *
     * @return
     */
    public String read() {
        String token;
        // 获取读锁（也可以尝试再次获取乐观读）
        long stamp = stampedLock.readLock();
        try {
            // 重新读取数据
            token = this.token;
        } finally {
            // 解锁读锁
            stampedLock.unlockRead(stamp);
        }
        return token;
    }

    /**
     * 锁升级
     */
    public void upgrade(String token) {
        long stamp = stampedLock.readLock();
        long l = stampedLock.tryConvertToWriteLock(stamp);
        // l 不为 0，锁升级成功
        if (l != 0) {
            this.token = token;
        }
    }
