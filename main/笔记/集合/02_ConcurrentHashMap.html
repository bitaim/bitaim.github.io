# ConcurrentHashMap

# 7和8的区别

  在JDK7中，ConcurrentHashMap使用的是 Segment + HashEntry + ReentrantLock的方式，采用分段锁的思想。

  在JDK8中，ConcurrentHashMap使用的是 Node + CAS + synchronized的方式。并使用红黑树来优化链表。

  JDK8的实现降级了锁的粒度，8是基于每一个桶加锁，7是对Segment段加锁。

# volatile关键字

  在其中使用了大量的volatile关键字，来保证数据的可见性。比如底层的Node数组，来保证扩容时的可见性。Node节点中 val和next都被修饰，保证能获取最新的数据，求他线程的修改能被及时发现。

final V putVal(K key, V value, boolean onlyIfAbsent) {
        if (key == null || value == null) throw new NullPointerException();
        // 计算哈希值 
        int hash = spread(key.hashCode());
        int binCount = 0;
        for (Node<K,V>[] tab = table;;) {
            Node<K,V> f; int n, i, fh;
            // 底层数组为空
            if (tab == null || (n = tab.length) == 0)
                // 初始化底层数组
                tab = initTable();
            // 数组[下标] 对应的节点为null
            else if ((f = tabAt(tab, i = (n - 1) & hash)) == null) {
                // CAS修改 tab[i] 的节点，由 null -> 新节点
                if (casTabAt(tab, i, null,
                             new Node<K,V>(hash, key, value, null)))
                    // 修改成功，则跳出循环
                    break;                   // no lock when adding to empty bin
            }
            // 当前节点正在执行扩容
            else if ((fh = f.hash) == MOVED)
                // 当前线程帮助去完成扩容
                tab = helpTransfer(tab, f);
            else {
                V oldVal = null;
                synchronized (f) {
                    if (tabAt(tab, i) == f) {
                        // 链表结构的插入
                        if (fh >= 0) {
                            binCount = 1;
                            for (Node<K,V> e = f;; ++binCount) {
                                K ek;
                                // 存在旧值，做覆盖
                                if (e.hash == hash &&
                                    ((ek = e.key) == key ||
                                     (ek != null && key.equals(ek)))) {
                                    oldVal = e.val;
                                    if (!onlyIfAbsent)
                                        e.val = value;
                                    break;
                                }
                                Node<K,V> pred = e;
                                
                                // 到链表尾部了
                                if ((e = e.next) == null) {
                                    // 插入到链表尾部
                                    pred.next = new Node<K,V>(hash, key,
                                                              value, null);
                                    break;
                                }
                            }
                        }
                        // 树形结构的插入
                        else if (f instanceof TreeBin) {
                            Node<K,V> p;
                            binCount = 2;
                            if ((p = ((TreeBin<K,V>)f).putTreeVal(hash, key,
                                                           value)) != null) {
                                oldVal = p.val;
                                if (!onlyIfAbsent)
                                    p.val = value;
                            }
                        }
                    }
                }
                if (binCount != 0) {
                    if (binCount >= TREEIFY_THRESHOLD)
                        treeifyBin(tab, i);
                    if (oldVal != null)
                        return oldVal;
                    break;
                }
            }
        }
        // 判断是否需要扩容
        addCount(1L, binCount);
        return null;
    }

static final int spread(int h) {
    // 进行了额外的与运算，保证哈希值为正数
    return (h ^ (h >>> 16)) & HASH_BITS;

# 3、初始化方法

private final Node<K,V>[] initTable() {
    Node<K,V>[] tab; int sc;
    // 如果底层的Node数组为null 或者 数组长度为0 代表还未初始化
    while ((tab = table) == null || tab.length == 0) {
        // 如果 sc 小于0 ，代表此时正有其他线程在执行初始化
        if ((sc = sizeCtl) < 0)
            // 交出CPU，陷入自旋
            // 当下一次进入while()时，由于tab是由volatile修饰的
            // 每次都会在主存中获取到最新的数据，而不必担心可见性问题
            Thread.yield(); // lost initialization race; just spin
        // 初始化底层Node数组
        // 此处使用CAS技术
        // this代表当前ConcurrentHashMap对象
        // SIZECTL 是 sizeCtl 在ConcurrentHashMap对象中的内存偏移量
        // CAS 将 sc = 0 修改为 -1
        else if (U.compareAndSwapInt(this, SIZECTL, sc, -1)) {
            try {
                if ((tab = table) == null || tab.length == 0) {
                    int n = (sc > 0) ? sc : DEFAULT_CAPACITY;
                    @SuppressWarnings("unchecked")
                    Node<K,V>[] nt = (Node<K,V>[])new Node<?,?>[n];
                    table = tab = nt;
                    // 默认初始化为 16  则sc = 16 - 4 = 12
                    // sc = n - (n / 4) = 3/4 n
                    // 永远是0.75n 这里应该是负载因子参数，但是没有使用到
                    // sc 是计算 底层数组的可用大小
                    sc = n - (n >>> 2);
                }
            } finally {
                sizeCtl = sc;
            }
            break;
        }
    }
    return tab;
}

# 1、扩容判断

private final void addCount(long x, int check) {
    CounterCell[] as; long b, s;
    
    // 记录元素的数量
    // （1）如果 计数单元数组 还没有初始化，代表并发竞争小。尝试通过CAS更新基础值。
    // （1.1）CAS修改成功，则跳过 if代码块。
    // （1.2）CAS修改失败，进入if代码块。初始化计数单元，并记录数量。
    // （2）计数单元数组 已经初始化，基于 计数单元数组 记录数量。
    if ((as = counterCells) != null || !U.compareAndSwapLong(this, BASECOUNT, b = baseCount, s = b + x)) {

        CounterCell a; long v; int m;

        // 没有发生竞争标志位
        boolean uncontended = true;

        // （1）as == null || (m = as.length - 1) < 0 ，计数单元数组还没有初始化
        // （2）计数单元数组已初始化，随机获取一个计数单元如果，但是为null
        // （3）随机获取的计数单元不为null，通过CAS修改 计数单元 的数量失败
        if (as == null || (m = as.length - 1) < 0 || 
            (a = as[ThreadLocalRandom.getProbe() & m]) == null || !(uncontended = U.compareAndSwapLong(a, CELLVALUE, v = a.value, v + x))) {

            // 尽全力记录元素数量
            fullAddCount(x, uncontended);
            // 直接返回了，可能是因为竞争这么大了，你们这些线程就别参与扩容了。
            return;

        }

        // 两种无须检查扩容的情况
        // （1）check = 1的情况，当插入新元素时，发现该位置已有数据。遍历链表时，发现第一个元素与新元素相同，那么就是覆盖操作。
        // （2）check = 0的情况，当插入新元素时，发现该位置没有数据。直接通过CAS更新该桶。

        // 番外：check > 1，该桶为树节点，或者最少遍历到链条的第二个节点。
        if (check <= 1)
            return;

        // 计算所有元素的数量
        s = sumCount();

    }


    // check：是否需要扩容检查，大于等于0就需要检查，默认为0
    if (check >= 0) {
        Node<K,V>[] tab, nt; int n, sc;
        
        //  扩容的三个条件
        // （1）当前元素数量 大于等于 sizeCtl
        // （2）底层数组不为空
        // （3）底层数组容量小于最大容量
        while (s >= (long)(sc = sizeCtl) && (tab = table) != null && (n = tab.length) < MAXIMUM_CAPACITY) {
            
            // 返回一个扩容戳
            int rs = resizeStamp(n);
            
            // 正在扩容中
            if (sc < 0) {
                // 五种取消扩容的情况
                // （1）sc 右移 16位，高位的扩容戳不同
                // （2）JDK8的bug（sc == ( rs <<< RESIZE_STAMP_SHIFT ) +1）。低16位为1，代表所有线程已经完成扩容任务
                // （3）JDK8的bug（sc == ( rs <<< RESIZE_STAMP_SHIFT ) + MAX_RESIZERS）。已达到协助扩容线程上限（65535）
                // （4）扩容已经完成
                // （5）所有的扩容任务已经分配完毕
                if ((sc >>> RESIZE_STAMP_SHIFT) != rs || sc == rs + 1 ||
                    sc == rs + MAX_RESIZERS || (nt = nextTable) == null ||
                    transferIndex <= 0)
                    break;
                // 将协助扩容的线程数量 加一
                if (U.compareAndSwapInt(this, SIZECTL, sc, sc + 1))
                    // 协助扩容
                    transfer(tab, nt);
            }
            
            // 还没执行扩容，第一个来扩容的线程的情况
            // 通过CAS将 sizeCtl 更新为负数（具体看下一代码块）
            else if (U.compareAndSwapInt(this, SIZECTL, sc,  (rs << RESIZE_STAMP_SHIFT) + 2)) {
                // 扩容
                transfer(tab, null);
            }
                        
            s = sumCount();
        }
    }

}

# 2、执行扩容


/**
 * tab：原数组
 * nextTab：新数组
 * transfer方法：负责将元素从原数组转移到新数组
 */
private final void transfer(Node<K,V>[] tab, Node<K,V>[] nextTab) {

        // 原数组的长度
        int n = tab.length;
        // 每个线程要处理的桶的数量
        int stride;
        
        // （1.1）如果CPU核心数 > 1，用原数组长度 / 8 / CPU核心数，将结果赋值给 stride。
        // （1.2）如果CPU核心数 = 1，直接将原数组长度赋值给 stride。
        // （1.3）如果 stride < 16，则赋予默认最小值16
        // 在（1.1）的情况下，当桶个数为512时，每个线程要处理的桶的个数还是16个。大部分情况下，stride 为 16。
        if ((stride = (NCPU > 1) ? (n >>> 3) / NCPU : n) < MIN_TRANSFER_STRIDE) {
            stride = MIN_TRANSFER_STRIDE;
        }
            
        // 新数组还未初始化
        if (nextTab == null) {          
            try {
                // 初始化新数组
                Node<K,V>[] nt = (Node<K,V>[])new Node<?,?>[n << 1];
                nextTab = nt;
            } catch (Throwable ex) {      // try to cope with OOME
                // 初始化新数组时，发生内存溢出，这里进行特殊处理，避免系统崩溃
                // sizeCtl 赋予最大值，不允许再次扩容，后续的读写基于旧数组
                sizeCtl = Integer.MAX_VALUE;
                return;
            }
            // 将新数组赋值给 成员变量nextTable
            nextTable = nextTab;
            
            // 将旧的数组长度赋值给 成员变量transferIndex ，用于给线程划分桶区间
            transferIndex = n;
        }
        // 新数组的长度
        int nextn = nextTab.length;
        
        // 创建 ForwardingNode 节点对新数组进行包装，fwd负责在旧数组中进行占位（该桶的元素转移成功后，会将fwd放入当前桶）
        // 当其他线程调用get方法获取数据时，如果发现hash值为-1，代表当前数组正在发生扩容，则线程会尝试帮助扩容
        ForwardingNode<K,V> fwd = new ForwardingNode<K,V>(nextTab);
        
        // 转移一个桶的完成标志位
        // 默认为true，是为了分配一段桶区间
        boolean advance = true;
        
        // 扩容完成标志位（旧数组中所有桶均已转移完毕）
        boolean finishing = false;
        
        // i：要处理的桶的下标
        // bound： 表示当前线程要处理桶的最小下标
        for (int i = 0, bound = 0;;) {
            Node<K,V> f; int fh;
            
            while (advance) {
                // nextIndex：要处理的桶下标（从桶区间的最大值倒序处理）
                // nextBound：当前线程要处理桶的区间最小值
                int nextIndex, nextBound;
                
                // （1）如果 --i >= bound 为true，代表当前线程被分配了桶区间，当前正在执行扩容操作。转移了15，-- 该转移14了。
                // （2）如果 --i >= bound 为false，则判断是否完成了。
                // （2.1）未完成，进入 else if ((nextIndex = transferIndex) <= 0)
                // （2.2）已完成，
                if (--i >= bound || finishing) 
                    advance = false;
                    
                // 获取 transferIndex 的值，赋值给 nextIndex（一个线程完成了桶的转移任务，还有机会获取到新任务）
                // 如果 nextIndex 小于等于0，代表区间已经被分配完毕
                else if ((nextIndex = transferIndex) <= 0) {
                    i = -1;
                    advance = false;
                }
                // 分配桶区间
                else if (U.compareAndSwapInt
                        // 通过CAS更新 transferIndex的值 到分配区间的最小值
                         (this, TRANSFERINDEX, nextIndex, nextBound = 
                            // 计算桶区间最小值。比如桶为16时，区间最小值为0。桶为32时，区间最小值为16。
                            // 分配桶区间时，第一个线程是[16,32)，第二个线程是[0,16)，从大端到小端。
                            (nextIndex > stride ?  nextIndex - stride : 0)
                            )) {
                            
                    // 桶区间的赋值
                    bound = nextBound;
                    
                    // 当前线程要转移的桶的下标。减一，是在大端到下端进行转移。比如分配到的区间是[0,16)，第一个要转移的桶为15。
                    i = nextIndex - 1;
                    advance = false;
                }
            }
            
            // 这里的三个条件，只有第一个可以成功。第二、第三个条件，不知道为啥这样写，根本无法满足
            // 假设原数组长度为32，需要有两个线程进行扩容。线程1分配[16,32)，线程2分配[0,16)。
            // 虽然线程1先执行，但是线程2先完成了并且结束返回了。线程1还需要额外进行[0,15)的循环，直到i<0才可以进入该代码块。
            if (i < 0 || i >= n || i + n >= nextn) {
                int sc;
                // 扩容完成标志位为true
                if (finishing) {
                    // 清空扩容的数组指针
                    nextTable = null;
                    // 指向新数组
                    table = nextTab;
                    // 计算下一次的扩容阈值（0.75 * n）。比如16扩容到32时，阈值为24。
                    sizeCtl = (n << 1) - (n >>> 1);
                    return;
                }
                // 通过CAS尝试将 sc - 1，将协助线程数减1
                if (U.compareAndSwapInt(this, SIZECTL, sc = sizeCtl, sc - 1)) {
                    // 不相等，表示扩容未结束
                    if ((sc - 2) != resizeStamp(n) << RESIZE_STAMP_SHIFT)
                        // 当前线程已完成桶的转移任务，结束扩容方法
                        return;
                    // 扩容结束，设置标志位
                    finishing = advance = true;
                    // 重新检查旧数组的所有桶
                    i = n; // recheck before commit
                }
            }
            // 原数组的i位置为null（该位置上没有元素需要转移到新数组）
            else if ((f = tabAt(tab, i)) == null)
                // 直接用 fwd 进行占位，代表已经转移成功
                advance = casTabAt(tab, i, null, fwd);
            // hash值为 MOVED 代表已经处理过（已经转移成功了）
            else if ((fh = f.hash) == MOVED)
                advance = true; // already processed
            else {
                // 具体的转移逻辑，链表拆分为两个链表转移到新数组
                // 利用fwd进行占位
                synchronized (f) {
                    if (tabAt(tab, i) == f) {
                        Node<K,V> ln, hn;

                        // 该桶为链表
                        if (fh >= 0) {

                            // 原哈希值与原数组长度进行与运算，runBit有两种情况
                            // （1）0。链表的头节点属于 低位节点，新下标 保持原桶的下标。
                            // （2）大于0。链表的头节点属于 高位节点，新下标 = 原桶的下标 + n。
                            int runBit = fh & n;

                            Node<K,V> lastRun = f;

                            for (Node<K,V> p = f.next; p != null; p = p.next) {
                                int b = p.hash & n;
                                if (b != runBit) {
                                    runBit = b;
                                    lastRun = p;
                                }
                            }
                            if (runBit == 0) {
                                ln = lastRun;
                                hn = null;
                            }
                            else {
                                hn = lastRun;
                                ln = null;
                            }
                            for (Node<K,V> p = f; p != lastRun; p = p.next) {
                                int ph = p.hash; K pk = p.key; V pv = p.val;
                                if ((ph & n) == 0)
                                    ln = new Node<K,V>(ph, pk, pv, ln);
                                else
                                    hn = new Node<K,V>(ph, pk, pv, hn);
                            }
                            setTabAt(nextTab, i, ln);
                            setTabAt(nextTab, i + n, hn);
                            setTabAt(tab, i, fwd);
                            advance = true;
                        }
                        // 该桶为树节点
                        else if (f instanceof TreeBin) {
                            TreeBin<K,V> t = (TreeBin<K,V>)f;
                            TreeNode<K,V> lo = null, loTail = null;
                            TreeNode<K,V> hi = null, hiTail = null;
                            int lc = 0, hc = 0;
                            for (Node<K,V> e = t.first; e != null; e = e.next) {
                                int h = e.hash;
                                TreeNode<K,V> p = new TreeNode<K,V>
                                    (h, e.key, e.val, null, null);
                                if ((h & n) == 0) {
                                    if ((p.prev = loTail) == null)
                                        lo = p;
                                    else
                                        loTail.next = p;
                                    loTail = p;
                                    ++lc;
                                }
                                else {
                                    if ((p.prev = hiTail) == null)
                                        hi = p;
                                    else
                                        hiTail.next = p;
                                    hiTail = p;
                                    ++hc;
                                }
                            }
                            ln = (lc <= UNTREEIFY_THRESHOLD) ? untreeify(lo) :
                                (hc != 0) ? new TreeBin<K,V>(lo) : t;
                            hn = (hc <= UNTREEIFY_THRESHOLD) ? untreeify(hi) :
                                (lc != 0) ? new TreeBin<K,V>(hi) : t;
                            setTabAt(nextTab, i, ln);
                            setTabAt(nextTab, i + n, hn);
                            setTabAt(tab, i, fwd);
                            advance = true;
                        }
                    }
                }
            }
        }
    }

# 3、resizeStamp(int n)方法

// RESIZE_STAMP_BITS初始值为16
static final int resizeStamp(int n) {
    // 高位到低位持续为0的个数 | 32768（0000 0000 0000 0000 1000 0000 0000 0000，1左移15位，也就是在1后面补15个0）
    return Integer.numberOfLeadingZeros(n) | (1 << (RESIZE_STAMP_BITS - 1));
}

  方法Integer.numberOfLeadingZeros(n)输入n为16时，二进制数字为0000 0000 0000 0000 0000 0000 0001 0000，则该方法返回27。

# 查询数据

public V get(Object key) {
    Node<K,V>[] tab; Node<K,V> e, p; int n, eh; K ek;
    // 计算哈希值（key不能为null，空指针异常）
    int h = spread(key.hashCode());
    // 确定底层数组已初始化，key所在的数组下标对应的节点不为空
    if ((tab = table) != null && (n = tab.length) > 0 &&
        (e = tabAt(tab, (n - 1) & h)) != null) {
        // 头节点的哈希值 与 要查询key的哈希值相同
        if ((eh = e.hash) == h) {
            // 头节点的key与要查询的key的地址相同，或者两个对象相等
            if ((ek = e.key) == key || (ek != null && key.equals(ek)))
                // 返回头节点的value值
                return e.val;
        }
        // 元素的哈希值小于0
        else if (eh < 0)
            // eh = -1，代表正在扩容中，调用 ForwardingNode find()方法，去新数组中寻找
            // eh = -2，代表当前是一颗红黑树，调用 TreeBin find()方法，去树中寻找
            return (p = e.find(h, key)) != null ? p.val : null;
            
        // 遍历链表
        while ((e = e.next) != null) {
            if (e.hash == h &&
                ((ek = e.key) == key || (ek != null && key.equals(ek))))
                return e.val;
        }
    }
    // 1、数组未初始化
    // 2、数组[下标]的节点位null
    // 3、遍历链表没有找到数据
    return null;
}
