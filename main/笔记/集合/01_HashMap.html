# HashMap

  HashMap的底层数据结构是，数组 + 链表 + 红黑树。

# 插入数据

# 1、源码

public V put(K key, V value) {
    return putVal(hash(key), key, value, false, true);
}

final V putVal(int hash, K key, V value, boolean onlyIfAbsent,
               boolean evict) {
    Node<K,V>[] tab; Node<K,V> p; int n, i;
    // 哈希表未初始化的情况
    if ((tab = table) == null || (n = tab.length) == 0)
        // 触发扩容
        n = (tab = resize()).length;
        
    // 还未添加过该下标的节点
    if ((p = tab[i = (n - 1) & hash]) == null)
        tab[i] = newNode(hash, key, value, null);
    else {
        // 当该key已经被存储了 就会找到该节点 赋值给e
        Node<K,V> e; K k;
        // p 是Node链表的第一个元素
        // 这里快速判断第一个元素是否一样
        // 哈希值相同 && （地址相同 或者 值相同）
        // 如果是自定义对象作为key，一定要重写equals()、hashCode()方法
        if (p.hash == hash &&
            ((k = p.key) == key || (key != null && key.equals(k))))
            e = p;
        else if (p instanceof TreeNode)
            e = ((TreeNode<K,V>)p).putTreeVal(this, tab, hash, key, value);
        else {
            for (int binCount = 0; ; ++binCount) {
                // 这里判断该下标bin的第二个节点 第一个已经在第一个if处判断过了
                if ((e = p.next) == null) {
                    p.next = newNode(hash, key, value, null);
                    if (binCount >= TREEIFY_THRESHOLD - 1) // -1 for 1st
                        treeifyBin(tab, hash);
                    break;
                }
                // 第二个节点相同 则跳过
                if (e.hash == hash &&
                    ((k = e.key) == key || (key != null && key.equals(k))))
                    break;
                // 指针指向下面的节点 ，既然遍历
                p = e;
            }
        }
        // 存在key的映射节点
        // 替换原本节点的value
        if (e != null) { // existing mapping for key
            V oldValue = e.value;
            if (!onlyIfAbsent || oldValue == null)
                e.value = value;
            afterNodeAccess(e);
            return oldValue;
        }
    }
    ++modCount;
    // 将长度加一，同时判断是否大于临界值
    if (++size > threshold)
        // 比如hashMap默认容量为16，当插入第13个元素后，就需要扩容了
        // 扩容
        resize();
    afterNodeInsertion(evict);
    return null;
}

# 2、哈希算法

  在JDK8时，将原始哈希值的高16位与低16位进行异或运算（相同时为0，不同时为1）。让最终结果保持高16位与低16位的特征，来减少哈希碰撞。代码如下：

static final int hash(Object key) {
    int h;
    return (key == null) ? 0 : (h = key.hashCode()) ^ (h >>> 16);
}

  异或运算过程如下。

哈希值  ：1111 1111 1111 1111 1111 1010 0111 1100
右移16位：0000 0000 0000 0000 1111 1111 1111 1111  把原hash的高16位推到了低16位，前面用0补齐
—————————————————————————————————————————————————
异或运算：1111 1111 1111 1111 0000 0101 1000 0011 -> int值

# 3、寻址算法

  将新的哈希值与（数组长度-1）进行与运算，计算出数组的下标。取代取模运算，其性能更高。

哈希值      ：1111 1111 1111 1111 0000 0101 1000 0011
数组长度-1  ：0000 0000 0000 0000 0000 0000 0000 1111
—————————————————————————————————————————————————————
与运算      ：0000 0000 0000 0000 0000 0000 0000 0011 -> 数组中的一个位置

# 4、哈希碰撞

  如果没有经过哈希算法的优化。假设两个key的哈希值低16位相同，高16位不同。当进行与运算时，计算出的数组下标是相同的。

  经过优化后，计算的最终结果保留高16位和低16位的特征。在进行与运算，减少出现同一位置的概率，也就是减少了hash碰撞的概率。

  经过哈希算法优化后，总归还是有概率落在同一位置上，此时就会拉出一个链表。

# 5、尾插法

  在JDK8之前，新元素会插入到链表的头部。而在JDK8开始，插入到链表的尾部。

  在JDK7时，扩容时会进行重哈希。因为采用的是头插法，改变了链表中元素的顺序。这样在多线程的情况下，可能产生环形链表，在查询时导致死循环。

  在JDK8时，采取尾插法，在扩容时会保持链表原本的顺序，避免了产生环形链表。

# 6、扩容时机

  HashMap的默认初始长度是16。当指定初始化大小时，HashMap会保证容量为2的幂次方。确保进行与运算时，（数组长度-1）的二进制数都为1。只要hashcode本身分布均匀，最终寻址的结果就是均匀的，实现均匀分布。

  HashMap的扩容，由两个参数影响。

    int Capacity; // HashMap当前长度
    static final float DEFAULT_LOAD_FACTOR = 0.75; // 负载因子，默认值0.75f。

  如果当前长度 > 数组长度 * DEFAULT_LOAD_FACTOR，就会触发resize扩容。

  扩容分两步，首先创建一个新的空数组，长度是原数组的2倍。然后rehash，重新对每个hash值进行寻址，也就是用每个hash值与新的数组的length-1进行与运算，重新计算在新数组中的位置。

  重哈希计算数组下标时，保证 原位置 或者 原数组大小 + 原下标。每次扩容，相对于原数组长度来看，在二进制数字上多了个1。

# 7、扩容源码

  当第一次添加元素时，由于底层哈希表还没有初始化。进行第一次初始化扩容。

final Node<K,V>[] resize() {
    // 获取当前哈希表
    Node<K,V>[] oldTab = table;
    // 计算当前哈希表长度
    int oldCap = (oldTab == null) ? 0 : oldTab.length;
    // threshold 初始化容量值
    // new HashMap(8),创建HashMap对象，指定容量时就会赋值给该变量
    int oldThr = threshold;
    int newCap, newThr = 0;
    if (oldCap > 0) {
        if (oldCap >= MAXIMUM_CAPACITY) {
            threshold = Integer.MAX_VALUE;
            return oldTab;
        }
        else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY &&
                 oldCap >= DEFAULT_INITIAL_CAPACITY)
            newThr = oldThr << 1; // double threshold
    }
    // 指定初始化容量时的扩容
    else if (oldThr > 0) // initial capacity was placed in threshold
        newCap = oldThr;
    else {               // zero initial threshold signifies using defaults
        // new HashMap() 未指定容量时，第一次初始化扩容
        // DEFAULT_INITIAL_CAPACITY 默认是16
        newCap = DEFAULT_INITIAL_CAPACITY; 
        // 计算下一次扩容的时机 = 0.75 * 16
        newThr = (int)(DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);
    }
    if (newThr == 0) {
        float ft = (float)newCap * loadFactor;
        newThr = (newCap < MAXIMUM_CAPACITY && ft < (float)MAXIMUM_CAPACITY ?
                  (int)ft : Integer.MAX_VALUE);
    }
    threshold = newThr;
    @SuppressWarnings({"rawtypes","unchecked"})
        // 创建新的Node数组
        Node<K,V>[] newTab = (Node<K,V>[])new Node[newCap];
    // 旧的哈希表指向新地址
    table = newTab;
    // 如果旧的哈希表不为空，则进行重哈希
    if (oldTab != null) {
        for (int j = 0; j < oldCap; ++j) {
            Node<K,V> e;
            // 获取bin
            if ((e = oldTab[j]) != null) {
                // 将原引用置为null
                oldTab[j] = null;
                // 该bin上只有一个节点
                if (e.next == null)
                    newTab[e.hash & (newCap - 1)] = e;
                // 该bin已经树化
                else if (e instanceof TreeNode)
                    ((TreeNode<K,V>)e).split(this, newTab, j, oldCap);
                // 遍历该链表，重哈希
                else { // preserve order
                    Node<K,V> loHead = null, loTail = null;
                    Node<K,V> hiHead = null, hiTail = null;
                    Node<K,V> next;
                    do {
                        // 拿到下一个节点
                        next = e.next;
                        if ((e.hash & oldCap) == 0) {
                            if (loTail == null)
                                loHead = e;
                            else
                                loTail.next = e;
                            loTail = e;
                        }
                        else {
                            if (hiTail == null)
                                hiHead = e;
                            else
                                hiTail.next = e;
                            hiTail = e;
                        }
                    } while ((e = next) != null);
                    if (loTail != null) {
                        loTail.next = null;
                        newTab[j] = loHead;
                    }
                    if (hiTail != null) {
                        hiTail.next = null;
                        newTab[j + oldCap] = hiHead;
                    }
                }
            }
        }
    }
    return newTab;
}

# 8、树化条件

/**
 * 当添加一个元素时，该bin上的节点至少有8个才会树化。
 */
static final int TREEIFY_THRESHOLD = 8;

/**
 * 在扩容时，如果该bin上的节点数量小于等于6时，将树化转为链表结构。
 */
static final int UNTREEIFY_THRESHOLD = 6;


/**
 * The smallest table capacity for which bins may be treeified.
 * (Otherwise the table is resized if too many nodes in a bin.)
 * Should be at least 4 * TREEIFY_THRESHOLD to avoid conflicts
 * between resizing and treeification thresholds.
 *
 * 只有哈希表容量大于等于64时，才可以进行树化。
 * 否则当一个bin上有太多的节点时，只会扩容。
 *
 */
static final int MIN_TREEIFY_CAPACITY = 64;

  链表最差的情况O(n)。当数组长度达到64，并且链表长度到达8时，会转化为红黑树 O(logn)。

  根据泊松分布，在负载因子默认为0.75时。单个槽内元素个数为8的概念小于百万分之一，所以将7作为一个分水岭。等于7的时候不转化，大于等于8才进行转换为红黑树，小于等于6才转化为链表。
