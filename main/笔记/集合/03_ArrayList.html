    # ArrayList

    # 源码分析

    1、变量

    // 当实例化ArrayList对象未指定容量，在第一次添加数据时，默认的扩容容量为10
private static final int DEFAULT_CAPACITY = 10;

// 底层通过 Object[]数组 存储数据
transient Object[] elementData;

// 当实例化 ArrayList 指定初始化容量为0时
// elementData = EMPTY_ELEMENTDATA;
private static final Object[] EMPTY_ELEMENTDATA = {};

// 当实例化ArrayList对象未指定容量时
// elementData = DEFAULTCAPACITY_EMPTY_ELEMENTDATA;
private static final Object[] DEFAULTCAPACITY_EMPTY_ELEMENTDATA = {};

// 元素的数量，而不是数组的长度
private int size;

    2、方法

    // 增加元素
public boolean add(E e) {
    // 确保数组容量（长度）足够
    ensureCapacityInternal(size + 1); 
    elementData[size++] = e;
    return true;
}

// 向数组中某个位置插入元素
public void add(int index, E element) {
    // 检查是否下标越界，允许的范围为[0,size]
    // （1）如果index = size，表示向最后一个位置插入元素，与add(E e)方法等价
    // （2）如果index != size，表示向数组中间插入元素，需要移动元素
    rangeCheckForAdd(index);

    // 确保数组容量（长度）足够
    ensureCapacityInternal(size + 1); 
    System.arraycopy(elementData, index, elementData, index + 1,
                     size - index);
    elementData[index] = element;
    size++;
}

// 替换数组中的某个元素
public E set(int index, E element) {
    // 检查是否下标越界，允许的范围为[0,size)
    rangeCheck(index);

    E oldValue = elementData(index);
    elementData[index] = element;
    return oldValue;
}

// 查找数组中的某个元素，因此index < size
public E get(int index) {
    // index < size，否则异常
    rangeCheck(index);

    return elementData(index);
}

    3、扩容

    // 确保数组容量足够
private void ensureCapacityInternal(int minCapacity) {
    ensureExplicitCapacity(calculateCapacity(elementData, minCapacity));
}


// 计算所需容量
private static int calculateCapacity(Object[] elementData, int minCapacity) {
    // 特例情况，未指定初始化容量，第一次扩容时的容量判断
    if (elementData == DEFAULTCAPACITY_EMPTY_ELEMENTDATA) {
        return Math.max(DEFAULT_CAPACITY, minCapacity);
    }
    return minCapacity;
}

private void ensureExplicitCapacity(int minCapacity) {
    modCount++;

    // 容量不足
    if (minCapacity - elementData.length > 0)
        // 扩容
        grow(minCapacity);
}


// 扩容
private void grow(int minCapacity) {
    // 当前容量，比如为10
    int oldCapacity = elementData.length;
    // 新容量，为15，在原容量的基础上增加一半
    int newCapacity = oldCapacity + (oldCapacity >> 1);
    // 扩容完了还不够，就按你的最低要求来
    if (newCapacity - minCapacity < 0)
        newCapacity = minCapacity;
    // 判断新容量是否溢出
    if (newCapacity - MAX_ARRAY_SIZE > 0)
        newCapacity = hugeCapacity(minCapacity);
    // 将旧数组拷贝到新数组中
    elementData = Arrays.copyOf(elementData, newCapacity);
}

    # fail-fast

    public class ArrayListTest {
    public static void main(String[] args) {
        List<String> messageList = new ArrayList<>();
        messageList.add("aa");
        messageList.add("bb");
        messageList.add("cc");

        // forEach(增强for循环) ，在编译后的字节码中使用的是Iterator迭代器
        for (String message : messageList) {}
        
        // Iterator迭代器进行遍历
        Iterator<String> iterator = messageList.iterator();
        while (iterator.hasNext()) {
            String message = iterator.next();
        }
    }
}

    1、什么是fail-fast

      当进行增强for循环时，会生成一个Iterator实例对象，同时将ArrayList的modCount变量赋值给Iterator实例对象的expectedModCount变量。具体代码如下。

    private class Itr implements Iterator<E> {
        int cursor;       // index of next element to return
        int lastRet = -1; // index of last element returned; -1 if no such
        // 赋值操作
        int expectedModCount = modCount;

        Itr() {}
}

      当在循环中调用ArrayList的写方法（通常也就是为了过滤集合中的数据）时，变量modCount就会自增。而每次迭代器循环一轮之后，就会验证expectedModCount与modCount是否相等，不等就抛出ConcurrentModificationException()并发修改异常。具体代码如下。

    public E next() {
    // 进行两个参数的校验
    checkForComodification();
    // ...略...
}

final void checkForComodification() {
    // 当修改次数 不等于 预期的修改次数时
    if (modCount != expectedModCount)
        throw new ConcurrentModificationException();
}

    2、如何避免fail-fast

    public class ArrayListTest {
    public static void main(String[] args) {
        List<String> messageList = new ArrayList<>();
        messageList.add("aa");
        messageList.add("bb");
        messageList.add("cc");

        while (iterator.hasNext()) {
            String next = iterator.next();
            if ("aa".equals(next)) {
                // 使用迭代器的移除方法，不要使用ArrayList的方法
                iterator.remove();
            }
        }
        
        // 在JDK1.8中，提供了简化的方法，是对上面代码的封装
        messageList.removeIf("aa"::equals);
    }
}

      迭代器的remove()方法具体代码如下。

    public void remove() {
    if (lastRet < 0)
        throw new IllegalStateException();
    checkForComodification();

    try {
        ArrayList.this.remove(lastRet);
        cursor = lastRet;
        lastRet = -1;
        // 将最新的modCount赋值给expectedModCount
        // 避免下一次循环时的校验错误
        expectedModCount = modCount;
    } catch (IndexOutOfBoundsException ex) {
        throw new ConcurrentModificationException();
    }
}

    3、为什么会有fail-fast

      在遍历期间尽可能多地检测干扰，而不会牺牲很多性能，主要依靠modCount遍历，但这些不能保证检测到并发冲突。取自ArrayList注释翻译。

      fail-fast 机制是java集合(Collection)中的一种错误机制。当多个线程对同一个集合进行操作时，就可能会产生fail-fast事件。例如：当某一个线程A通过iterator去遍历某集合的过程中，若该集合的数据被其他线程所改变了；那么线程A访问集合时，就会抛出ConcurrentModificationException异常，产生fail-fast事件。
