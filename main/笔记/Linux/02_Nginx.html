  • 5、在阿里云安全组开放443端口

2、修改Nginx配置

    server {
        listen       80;

		# begin 
        server_name  www.<主域名>.cn;
		rewrite ^/(.*) https://www.<主域名>.cn/$1 permanent;
		# end

        location / {
            root   html;
            index  index.html index.htm;
        }

        error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   html;
        }
    }
	
	# begin
	server {
		 listen 443;
		 server_name www.<主域名>.cn;
		 ssl on;
		 root html;
		 index index.html index.htm;
		 # 上传到服务器的密钥 Begin
		 ssl_certificate   /usr/local/https/www.<主域名>.cn.pem;
		 ssl_certificate_key  /usr/local/https/www.<主域名>.cn.key;
		 # 上传到服务器的密钥 End
		 ssl_session_timeout 5m;
		 ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECDHE:ECDH:AES:HIGH:!NULL:!aNULL:!MD5:!ADH:!RC4;
		 ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
		 ssl_prefer_server_ciphers on;
		 location / {
            root   html;
            index  index.html index.htm;
        }
	}
	# end

# 反向代理

http {


    # 开启Nginx的长连接
    keepalive_timeout 65;
    keepalive_requests 100;

    # tomcat 集群
    upstream tomcat_servers {
        # 失败三次，则触发熔断，10秒内不再分发请求给该服务器
        server 192.168.29.101:8080 weight=3 max_fails=3 fail_timeout=10s;
        server 192.168.29.102:8080 weight=2 max_fails=3 fail_timeout=10s;
        # 每个Worker可保持的长连接的最大空闲数
        # Tomcat的该值可以设置大一些
        keepalive 30;
    }
    server {
        port 80;

        proxy_read_timeout 3s; # 超时时间为3秒，默认是60秒。
        proxy_next_upstream_tries 2; # 重试次数
        proxy_next_upstream_timeout 6; # 重试的最长时间
        proxy_next_upstream error timeout; # 故障转移 http_503 http_500 http_404;
        

        location / {
            # 建立长连接，避免大量 time_wait
            proxy_http_version 1.1;
            proxy_set_header Connection "";

            
            # $host：nginx的ip
            # $http_host：nginx的ip和端口号
            # $proxy_host：反向代理到的ip地址
            # $proxy_port：反向代理到的端口号
            proxy_set_header Host $http_host:$proxy_port;

            # 获取用户的真实ip，下面是两种方式
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            
			# 错误拦截
            # proxy_intercept_errors on;

            # 反向代理
            proxy_pass http://tomcat_servers;
            
        }
    }
}

# 负载均衡策略

  • 1、轮询：将请求按顺序轮流地分配到后端服务器上，它均衡地对待后端的每一台服务器，而不关心服务器实际的连接数和当前的系统负载。
  • 2、加权轮询：不同的后端服务器可能机器的配置和当前系统的负载并不相同，因此它们的抗压能力也不相同。

  给配置高、负载低的机器配置更高的权重，让其处理更多的请求；而配置低、负载高的机器，给其分配较低的权重，降低其系统负载

  加权轮询能很好地处理这一问题，并将请求顺序且按照权重分配到后端。

  • 3、ip_hash（源地址哈希法）：根据获取客户端的IP地址，通过哈希函数计算得到一个数值，用该数值对服务器列表的大小进行取模运算，得到的结果便是客户端要访问服务器的序号。

  采用源地址哈希法进行负载均衡，同一IP地址的客户端，当后端服务器列表不变时，它每次都会映射到同一台后端服务器进行访问。

  • 4、随机：通过系统的随机算法，根据后端服务器的列表大小值来随机选取其中的一台服务器进行访问。
  • 5、least_conn（最小连接数法）：由于后端服务器的配置不尽相同，对于请求的处理有快有慢，最小连接数法根据后端服务器当前的连接情况，动态地选取其中当前积压连接数最少的一台服务器来处理当前的请求，尽可能地提高后端服务的利用效率，将负责合理地分流到每一台服务器。

# 反向代理故障转移

  Nginx将请求分配给其中一个实例失败时，比如该实例宕机了。Nginx会自动将请求转发给下一个实例，这叫做故障转移。但故障转移，同时也会发生在请求超时的场景下。

  Nginx中默认的请求超时时间为60秒。默认值太大了，通常会通过proxy_read_timeout 3s;参数进行调整。

  此时我们的实例A处理请求速度很慢，Nginx由于执行超时会将请求转发给下一个节点。如果所有的节点都超时，则返回超时响应给客户端。

  虽然响应给客户端请求超时，但其实请求仍然在后端服务器上执行着。这就可能导致服务器重复处理请求，因此需要保证幂等性。dubbo、springclooud等微服务架构内部调用的重试机制也会有这个问题。

  请求重试的问题。

  对于get请求。由于大多数get请求都是做数据查询，因此不会有业务数据的问题。在开发规范中，也就要要求get请求不能有写数据库的操作。

  对于post请求。当服务器宕机时，请求根本没有发出去，因为会进行故障转移进行重试。当出现超时时，Nginx并不会转发post请求到另外的服务器，避免了重复处理数据的问题。

# 解决跨域问题

server {
    listen 80;
    server_name localhost;
    add_header 'Access-Control-Allow-Origin' *;
    add_header 'Access-Control-Allow-Credentials' 'true';
    add_header 'Access-Control-Allow-Methods' *;
    add_header 'Access-Control-Allow-Headers' *;

    # 配置静态资源防盗链
    valid_referers *.imooc.com;
    if ($invalid_referer) {
        return 404;
    }
}

# 使用Gzip压缩

# 开启Gzip压缩功能，提高传输效率，节约带宽
gzip on;
# 限制最小压缩，小于1字节文件不会压缩
gzip_min_length 1;
# 压缩比，1~9之间，压缩比越大，CPU使用的也就越多
gzip_comp_level 3;
# 定义压缩文件的类型
gzip_types text/plain application/javascript application/x-javascript text/css application/xml text/javascript application/x-httpd-php image/jpeg image/gif image/png application/json;
