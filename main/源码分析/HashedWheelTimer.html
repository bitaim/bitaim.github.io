    # HashedWheelTimer

      HashedWheelTimer是Netty框架中的时间轮组件。

    # 快速开始

    import io.netty.util.HashedWheelTimer;
import io.netty.util.Timeout;
import io.netty.util.TimerTask;

import java.util.Set;
import java.util.concurrent.TimeUnit;

public class HashedWheelTimerDemo {

    /**
     * tickDuration：每个刻度的时间
     * unit：每个刻度的时间单位
     * ticksPerWheel：时间轮中刻度的数量，内部会将值向上转换为2的n次方, 10->16
     */
    private static HashedWheelTimer timer = new HashedWheelTimer(1, TimeUnit.SECONDS, 10);

    public static void main(String[] args) {

        TimerTask task = new TimerTask() {
            @Override
            public void run(Timeout timeout) throws Exception {
                // 任务执行应该单独开一个线程池，否则可能会导致时间轮延时异常
                System.out.println("任务开始执行");
                Thread.sleep(500);
                System.out.println("任务结束执行");

            }
        };

        // 添加5秒钟的延时任务
        Timeout timeout = timer.newTimeout(task, 5, TimeUnit.SECONDS);

        // 取消延时任务(不需要在具体的任务中判断任务状态, 任务被取消时时间轮不会调度该任务)
        timeout.cancel();

        // 停止时间轮, 返回未处理的非取消状态的任务
        Set<Timeout> unprocessedTasks = timer.stop();

    }

}

    # 构造方法

    public HashedWheelTimer(
        ThreadFactory threadFactory,
        long tickDuration, TimeUnit unit, int ticksPerWheel, boolean leakDetection,
        long maxPendingTimeouts) {

    ObjectUtil.checkNotNull(threadFactory, "threadFactory");
    ObjectUtil.checkNotNull(unit, "unit");
    ObjectUtil.checkPositive(tickDuration, "tickDuration");
    ObjectUtil.checkPositive(ticksPerWheel, "ticksPerWheel");

    // Normalize ticksPerWheel to power of two and initialize the wheel.
    // 时间轮大小 转换成 2的n次方，并创建时间轮
    wheel = createWheel(ticksPerWheel);
    // 时间轮大小-1 方便后面进行与运算计算下标
    mask = wheel.length - 1;

    // Convert tickDuration to nanos.
    long duration = unit.toNanos(tickDuration);

    // Prevent overflow.
    if (duration >= Long.MAX_VALUE / wheel.length) {
        throw new IllegalArgumentException(String.format(
                "tickDuration: %d (expected: 0 < tickDuration in nanos < %d",
                tickDuration, Long.MAX_VALUE / wheel.length));
    }

    if (duration < MILLISECOND_NANOS) {
        logger.warn("Configured tickDuration {} smaller then {}, using 1ms.",
                    tickDuration, MILLISECOND_NANOS);
        this.tickDuration = MILLISECOND_NANOS;
    } else {
        this.tickDuration = duration;
    }

    // 创建工作线程， 该线程负责时间轮刻度流转和到期任务的执行
    workerThread = threadFactory.newThread(worker);

    leak = leakDetection || !workerThread.isDaemon() ? leakDetector.track(this) : null;

    // 延时任务数量上限值，默认为-1不限制
    this.maxPendingTimeouts = maxPendingTimeouts;

    // 校验时间轮数量是否超过上限
    if (INSTANCE_COUNTER.incrementAndGet() > INSTANCE_COUNT_LIMIT &&
        WARNED_TOO_MANY_INSTANCES.compareAndSet(false, true)) {
        reportTooManyInstances();
    }
}

    # 添加任务

    public Timeout newTimeout(TimerTask task, long delay, TimeUnit unit) {
    ObjectUtil.checkNotNull(task, "task");
    ObjectUtil.checkNotNull(unit, "unit");

    // 延时任务数量加1
    long pendingTimeoutsCount = pendingTimeouts.incrementAndGet();

    // 判断延时任务数量是否超过上限， maxPendingTimeouts默认为-1，不进入校验逻辑
    if (maxPendingTimeouts > 0 && pendingTimeoutsCount > maxPendingTimeouts) {
        pendingTimeouts.decrementAndGet();
        throw new RejectedExecutionException("Number of pending timeouts ("
            + pendingTimeoutsCount + ") is greater than or equal to maximum allowed pending "
            + "timeouts (" + maxPendingTimeouts + ")");
    }

    // 时间轮工作线程开启
    start();

    // Add the timeout to the timeout queue which will be processed on the next tick.
    // During processing all the queued HashedWheelTimeouts will be added to the correct HashedWheelBucket.
    // 计算当前延时任务到期时间
    long deadline = System.nanoTime() + unit.toNanos(delay) - startTime;

    // Guard against overflow.
    if (delay > 0 && deadline < 0) {
        deadline = Long.MAX_VALUE;
    }

    // 延时任务添加到阻塞队列中，并没有直接放入时间轮对应的桶中
    HashedWheelTimeout timeout = new HashedWheelTimeout(this, task, deadline);
    timeouts.add(timeout);
    return timeout;
}


public void start() {
    // 判断工作线程状态
    switch (WORKER_STATE_UPDATER.get(this)) {
        // 如果是初始化状态，通过cas修改状态为开启，执行Worker线程
        case WORKER_STATE_INIT:
            if (WORKER_STATE_UPDATER.compareAndSet(this, WORKER_STATE_INIT, WORKER_STATE_STARTED)) {
                workerThread.start();
            }
            break;
        // 如果开启状态，直接跳过
        case WORKER_STATE_STARTED:
            break;
        // 如果关闭状态，抛出异常
        case WORKER_STATE_SHUTDOWN:
            throw new IllegalStateException("cannot be started once stopped");
        default:
            throw new Error("Invalid WorkerState");
    }

    // Wait until the startTime is initialized by the worker.
    // 等待工作线程初始化 时间轮开始时刻（内部判断任务到期使用的是相对时间，所以需要记录开始时刻）
    while (startTime == 0) {
        try {
            // 进入阻塞状态
            startTimeInitialized.await();
        } catch (InterruptedException ignore) {
            // Ignore - it will be ready very soon.
        }
    }
}

    # 工作线程Worker

    private final class Worker implements Runnable {
    private final Set<Timeout> unprocessedTimeouts = new HashSet<Timeout>();

    private long tick;

    @Override
    public void run() {
        // Initialize the startTime.
        // 初始化 开始时刻
        startTime = System.nanoTime();
        if (startTime == 0) {
            // We use 0 as an indicator for the uninitialized value here, so make sure it's not 0 when initialized.
            startTime = 1;
        }

        // Notify the other threads waiting for the initialization at start().
        // 唤醒阻塞的线程
        startTimeInitialized.countDown();

        do {
            // 进入下一个刻度，返回的是 已经过去的时刻
            final long deadline = waitForNextTick();
            if (deadline > 0) {
                // 计算时间轮桶的下标
                int idx = (int) (tick & mask);
                // 处理取消的任务
                processCancelledTasks();
                // 获取当前时间轮桶
                HashedWheelBucket bucket =
                        wheel[idx];
                // 将未处理的任务转移到桶中
                transferTimeoutsToBuckets();
                // 判断任务是否到期
                bucket.expireTimeouts(deadline);
                // 下标增加（注意下标是一直递增的）
                tick++;
            }
        } while (WORKER_STATE_UPDATER.get(HashedWheelTimer.this) == WORKER_STATE_STARTED);

        // 到这里的逻辑就是时间轮关闭时，跳出上面的循环
        // Fill the unprocessedTimeouts so we can return them from stop() method.
        
        // 遍历桶，将桶中的任务放入 unprocessedTimeouts
        for (HashedWheelBucket bucket: wheel) {
            bucket.clearTimeouts(unprocessedTimeouts);
        }
        // 将未处理的任务放入 unprocessedTimeouts
        for (;;) {
            HashedWheelTimeout timeout = timeouts.poll();
            if (timeout == null) {
                break;
            }
            if (!timeout.isCancelled()) {
                unprocessedTimeouts.add(timeout);
            }
        }
        // 处理取消的任务
        processCancelledTasks();
    }

    private void transferTimeoutsToBuckets() {
        // transfer only max. 100000 timeouts per tick to prevent a thread to stale the workerThread when it just
        // adds new timeouts in a loop.
        // 便利未处理阻塞队列中的任务，最多取出100000万条
        for (int i = 0; i < 100000; i++) {
            HashedWheelTimeout timeout = timeouts.poll();
            if (timeout == null) {
                // all processed
                break;
            }

            // 任务被取消，跳过该任务
            if (timeout.state() == HashedWheelTimeout.ST_CANCELLED) {
                // Was cancelled in the meantime.
                continue;
            }

            // 计算任务需要等待多少个刻度
            long calculated = timeout.deadline / tickDuration;
            // 计算任务执行的轮次
            timeout.remainingRounds = (calculated - tick) / wheel.length;
            // 计算任务所在桶的下标
            final long ticks = Math.max(calculated, tick); // Ensure we don't schedule for past.
            int stopIndex = (int) (ticks & mask);

            // 将任务加入桶中，桶中的任务还是个双向队列
            HashedWheelBucket bucket = wheel[stopIndex];
            bucket.addTimeout(timeout);
        }
    }

    private void processCancelledTasks() {
        for (;;) {
            HashedWheelTimeout timeout = cancelledTimeouts.poll();
            if (timeout == null) {
                // all processed
                break;
            }
            try {
                // 将任务移除
                timeout.remove();
            } catch (Throwable t) {
                if (logger.isWarnEnabled()) {
                    logger.warn("An exception was thrown while process a cancellation task", t);
                }
            }
        }
    }

    /**
     * calculate goal nanoTime from startTime and current tick number,
     * then wait until that goal has been reached.
     * @return Long.MIN_VALUE if received a shutdown request,
     * current time otherwise (with Long.MIN_VALUE changed by +1)
     */
    private long waitForNextTick() {
        // 计算进入下一个刻度需要的时间
        long deadline = tickDuration * (tick + 1);

        for (;;) {
            // 当前已启动时间
            final long currentTime = System.nanoTime() - startTime;
            // 计算这之间的差额
            long sleepTimeMs = (deadline - currentTime + 999999) / 1000000;
            // 小于0代表已满足进入下一刻度的条件
            if (sleepTimeMs <= 0) {
                if (currentTime == Long.MIN_VALUE) {
                    return -Long.MAX_VALUE;
                } else {
                    // 返回 当前已启动时间
                    return currentTime;
                }
            }

            // Check if we run on windows, as if thats the case we will need
            // to round the sleepTime as workaround for a bug that only affect
            // the JVM if it runs on windows.
            //
            // See https://github.com/netty/netty/issues/356
            if (PlatformDependent.isWindows()) {
                sleepTimeMs = sleepTimeMs / 10 * 10;
                if (sleepTimeMs == 0) {
                    sleepTimeMs = 1;
                }
            }

            // 休眠一会，等待进入下一刻度的时间到来
            try {
                Thread.sleep(sleepTimeMs);
            } catch (InterruptedException ignored) {
                if (WORKER_STATE_UPDATER.get(HashedWheelTimer.this) == WORKER_STATE_SHUTDOWN) {
                    return Long.MIN_VALUE;
                }
            }
        }
    }

    public Set<Timeout> unprocessedTimeouts() {
        return Collections.unmodifiableSet(unprocessedTimeouts);
    }
}

    # 任务到期执行

    
/**
 * HashedWheelBucket#expireTimeouts
 * Expire all {@link HashedWheelTimeout}s for the given {@code deadline}.
 */
public void expireTimeouts(long deadline) {
    // 获取桶中的头节点
    HashedWheelTimeout timeout = head;

    // process all timeouts
    // 桶中有任务
    while (timeout != null) {
        // 提前获取下一个任务
        HashedWheelTimeout next = timeout.next;
        // 任务执行轮次小于等于0代表可以执行了
        if (timeout.remainingRounds <= 0) {
            // 移除当前任务
            next = remove(timeout);
            // 任务到期了可以执行
            if (timeout.deadline <= deadline) {
                timeout.expire();
            } else {
                // The timeout was placed into a wrong slot. This should never happen.
                throw new IllegalStateException(String.format(
                        "timeout.deadline (%d) > deadline (%d)", timeout.deadline, deadline));
            }
            // 任务取消直接移除任务
        } else if (timeout.isCancelled()) {
            next = remove(timeout);
        } else {
            // 将任务轮次减1
            timeout.remainingRounds --;
        }
        // 切换任务指针
        timeout = next;
    }
}
